    ```js
      const color = document.getElementById('subprofile-color').value || '#f2a365';
      if (!name) { flashToast('Name required', { kind: 'warn' }); return; }
    ```

    ```js
    function renderOwnerSettings() {
      const isOwner = state.auth && state.ownerUid && state.auth.uid === state.ownerUid;
      const el = document.getElementById('settings-owner');
      if (el) el.style.display = isOwner ? 'block' : 'none';
      if (!isOwner) return;
      // Populate transfer-target dropdown with all authed members (not sub-profiles, not guests)
      const candidates = state.members.filter(m => m.uid && m.uid !== state.auth.uid && !m.temporary);
      const sel = document.getElementById('settings-transfer-target');
      if (sel) sel.innerHTML = candidates.map(m => `<option value="${m.uid}">${escapeHtml(m.name)}</option>`).join('');
    }
    // Call renderOwnerSettings() from wherever the settings screen opens/refreshes.
    // Also hook into the family-doc snapshot: when state.ownerUid changes, re-render.

    window.submitGroupPassword = async function() {
      const pw = (document.getElementById('settings-password-input').value || '').trim();
      if (pw && pw.length < 4) { flashToast('Password must be at least 4 characters', { kind: 'warn' }); return; }
      if (!state.auth || !state.familyCode) return;
      try {
        const fn = httpsCallable(functions, 'setGroupPassword');
        const r = await fn({ familyCode: state.familyCode, newPassword: pw });
        if (r.data && r.data.ok) {
          flashToast(pw ? 'Password set' : 'Password cleared', { kind: 'success' });
          haptic('success');
          document.getElementById('settings-password-input').value = '';
        }
      } catch(e) {
        console.error('[set-password]', e);
        const msg = (e.code === 'permission-denied') ? 'Only the owner can do this.' : "Couldn't save. Try again?";
        flashToast(msg, { kind: 'warn' });
      }
    };

    window.createGuestInvite = async function() {
      if (!state.auth || !state.familyCode) return;
      const duration = parseInt(document.getElementById('settings-guest-duration').value, 10);
      try {
        const fn = httpsCallable(functions, 'inviteGuest');
        const r = await fn({ familyCode: state.familyCode, durationMs: duration });
        if (r.data && r.data.ok) {
          const link = r.data.deepLink || `https://couchtonight.app/?invite=${r.data.inviteToken}&family=${state.familyCode}`;
          const out = document.getElementById('settings-guest-link-out');
          out.innerHTML = `<p>Send this link to your guest:</p><code>${escapeHtml(link)}</code><button onclick="copyToClipboard('${link}')">Copy</button>`;
          haptic('success');
        }
      } catch(e) {
        console.error('[invite-guest]', e);
        flashToast("Couldn't generate invite.", { kind: 'warn' });
      }
    };

    window.copyToClipboard = async function(text) {
      try { await navigator.clipboard.writeText(text); flashToast('Copied', { kind: 'success' }); }
      catch(e) { flashToast('Select and copy manually', { kind: 'info' }); }
    };

    window.transferOwnershipTo = async function() {
      const sel = document.getElementById('settings-transfer-target');
      const newOwnerUid = sel && sel.value;
      if (!newOwnerUid) { flashToast('Pick a member first', { kind: 'warn' }); return; }
      if (!confirm('Transfer ownership? You can ask for it back but the new owner has to agree.')) return;
      try {
        const fn = httpsCallable(functions, 'transferOwnership');
        const r = await fn({ familyCode: state.familyCode, newOwnerUid });
        if (r.data && r.data.ok) {
          flashToast('Ownership transferred', { kind: 'success' });
          haptic('success');
          // The family-doc snapshot (next task) will update state.ownerUid and re-render.
        }
      } catch(e) {
        console.error('[transfer-ownership]', e);
        const msg = (e.code === 'failed-precondition') ? 'That member has no account yet.' :
                    (e.code === 'permission-denied') ? 'Only the current owner can transfer.' :
                    "Couldn't transfer. Try again?";
        flashToast(msg, { kind: 'warn' });
      }
    };
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Owner-only UI ↔ client logic | Client hides owner panel when `state.auth.uid !== state.ownerUid`; CFs independently re-verify server-side |
| Sub-profile `managedBy` ↔ Firestore rules | Rules check `managedBy == uid()` when a write carries `managedMemberId` |
| Guest invite link ↔ sessionStorage → joinGroup CF | Token persists across sign-in redirect via sessionStorage stash (Plan 06 Step C) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-07-01 | Elevation-of-privilege | Non-owner clicks owner-only CF in DevTools | mitigate | CFs re-verify `ownerUid === uid`; client UI hide is convenience, not security |
| T-05-07-02 | Tampering | User forges `managedMemberId` in writeAttribution for a sub-profile they don't own | mitigate | Rules enforce `members/$(managedMemberId).data.managedBy == uid()` (Plan 04) |
| T-05-07-03 | Info-disclosure | Guest invite link shared with wrong person | mitigate | Single-use token (CF consumeInviteToken transaction); TTL bounded (1d/1w/30d max); rule-protected invites subcollection |
| T-05-07-04 | Tampering | Owner transfer to sub-profile (no uid) | mitigate | `transferOwnership` CF checks target has `uid` field; throws failed-precondition |
| T-05-07-05 | Availability | Guest's expired session still votes/vetoes | mitigate | Name-screen filter excludes expired guests; rules (Plan 04 post-grace path) deny writes from members without uid — sub-profiles unaffected because they route through act-as |
| T-05-07-06 | Spoofing | Act-as state leaks into unrelated action | mitigate | writeAttribution snapshot-then-clear (Plan 05) ensures per-action semantics |
| T-05-07-07 | Info-disclosure | Password rotation leaves old CF-minted invites valid | mitigate | `setGroupPassword` CF side-effect invalidates pending invites (status: invalidated) — Plan 03 |
| T-05-07-08 | Repudiation | Ownership-transfer without audit | mitigate | CF stamps `previousOwnerUid + ownershipTransferredAt` on family doc |
</threat_model>
