    ```js
    // js/auth.js — Firebase Auth provider wrapper (Phase 5)
    import {
      auth,
      GoogleAuthProvider, OAuthProvider,
      signInWithRedirect, getRedirectResult,
      signInWithPhoneNumber, RecaptchaVerifier,
      sendSignInLinkToEmail, isSignInWithEmailLink, signInWithEmailLink,
      onAuthStateChanged, firebaseSignOut
    } from './firebase.js';
    import { state } from './state.js';

    // ---- boot ---------------------------------------------------------------
    // Call ONCE at app boot, BEFORE any UI renders, so getRedirectResult can resolve
    // the return-from-provider hop before the router decides which screen to show.
    export async function bootstrapAuth() {
      // Restore intended post-sign-in destination if we stashed one before redirect.
      let freshFromRedirect = false;
      let redirectUser = null;
      try {
        const result = await getRedirectResult(auth);
        if (result) { freshFromRedirect = true; redirectUser = result.user; }
      } catch (e) {
        console.error('[auth] getRedirectResult failed', e);
      }
      // Snapshot the current user synchronously onto state so the first render is auth-aware.
      state.auth = auth.currentUser;
      return { freshFromRedirect, user: redirectUser || auth.currentUser };
    }

    export function watchAuth(cb) { return onAuthStateChanged(auth, cb); }

    // ---- providers ----------------------------------------------------------
    // Persist intended destination BEFORE redirect (RESEARCH.md §1 "Lost next param").
    function stashNextForRedirect() {
      try {
        const params = new URLSearchParams(window.location.search);
        if (params.get('claim')) sessionStorage.setItem('qn_claim', params.get('claim'));
        if (params.get('family')) sessionStorage.setItem('qn_claim_family', params.get('family'));
        if (params.get('invite')) sessionStorage.setItem('qn_invite', params.get('invite'));
      } catch(e) {}
    }

    export async function signInWithGoogle() {
      stashNextForRedirect();
      const provider = new GoogleAuthProvider();
      await signInWithRedirect(auth, provider);
    }

    export async function signInWithApple() {
      stashNextForRedirect();
      const provider = new OAuthProvider('apple.com');
      provider.addScope('email');
      provider.addScope('name');
      provider.setCustomParameters({ locale: (navigator.language || 'en') });
      await signInWithRedirect(auth, provider);
    }

    // Email link
    export async function sendEmailLink(email) {
      try { sessionStorage.setItem('qn_email_for_link', email); } catch(e) {}
      const actionCodeSettings = {
        url: 'https://couchtonight.app/?emaillink=1',
        handleCodeInApp: true
      };
      await sendSignInLinkToEmail(auth, email, actionCodeSettings);
    }

    export async function completeEmailLinkIfPresent() {
      if (!isSignInWithEmailLink(auth, window.location.href)) return null;
      let email = null;
      try { email = sessionStorage.getItem('qn_email_for_link'); } catch(e) {}
      if (!email) email = window.prompt('Confirm the email you used to sign in');
      if (!email) return null;
      const result = await signInWithEmailLink(auth, email, window.location.href);
      try { sessionStorage.removeItem('qn_email_for_link'); } catch(e) {}
      return result.user;
    }

    // Phone + invisible reCAPTCHA
    let recaptchaVerifier = null;
    export function initPhoneCaptcha(buttonElementId) {
      if (recaptchaVerifier) return recaptchaVerifier;
      recaptchaVerifier = new RecaptchaVerifier(auth, buttonElementId, {
        size: 'invisible',
        callback: () => {},
        'expired-callback': () => { recaptchaVerifier = null; }
      });
      return recaptchaVerifier;
    }
    export function resetPhoneCaptcha() {
      try { if (recaptchaVerifier) recaptchaVerifier.clear(); } catch(e) {}
      recaptchaVerifier = null;
    }
    export async function sendPhoneCode(phoneE164, buttonElementId) {
      const verifier = initPhoneCaptcha(buttonElementId);
      return await signInWithPhoneNumber(auth, phoneE164, verifier);
      // caller holds the ConfirmationResult and calls .confirm(code) with the SMS code.
    }

    // Sign-out
    export async function signOutUser() {
      await firebaseSignOut(auth);
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser ↔ Firebase Auth backend | Auth SDK makes HTTPS calls to Google; response carries an ID token |
| authDomain host (`couchtonight.app`) ↔ redirect handler | The `/__/auth/*` path is a Firebase Hosting reverse proxy onto Google's auth infra |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01-01 | Tampering | `firebaseConfig.authDomain` | mitigate | Value is source-controlled (public-by-design per CLAUDE.md). Any MITM rewriting the JS would need to breach TLS + Firebase Hosting — out of threat model scope for this phase |
| T-05-01-02 | Information-disclosure | Auth API key in source | accept | Firebase web API keys are scoped client-identifiers, not secrets (public-by-design). Rules + Auth-authorized-domains are the real gate |
| T-05-01-03 | Denial-of-service | Exposed `signInWithRedirect` kickoff | accept | Low value — redirects unauthenticated users to Google/Apple's own auth pages; no internal resource consumed until user actually completes sign-in |
| T-05-01-04 | Spoofing | Phishing site mimicking Couch sign-in | mitigate | authDomain locked to `couchtonight.app` in Auth Authorized Domains (configured in Plan 02) — Firebase rejects redirects to any other host |
</threat_model>
