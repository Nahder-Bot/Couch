    ```js
    const { onCall, HttpsError } = require('firebase-functions/v2/https');
    const admin = require('firebase-admin');
    const crypto = require('crypto');

    exports.mintClaimTokens = onCall({ region: 'us-central1', cors: true }, async (req) => {
      const { familyCode, memberIds, type } = req.data || {};
      const uid = req.auth?.uid;
      if (!uid) throw new HttpsError('unauthenticated', 'Sign in required');
      if (!familyCode || typeof familyCode !== 'string') throw new HttpsError('invalid-argument', 'familyCode required');
      if (!Array.isArray(memberIds) || memberIds.length === 0) throw new HttpsError('invalid-argument', 'memberIds array required');
      if (!['migration', 'graduation'].includes(type)) throw new HttpsError('invalid-argument', 'type must be migration or graduation');

      const famRef = admin.firestore().doc(`families/${familyCode}`);
      const fam = await famRef.get();
      if (!fam.exists) throw new HttpsError('not-found', 'Family not found');

      const isOwner = fam.data().ownerUid === uid;

      // For migration: caller must be the owner. For graduation: caller must be the parent (managedBy == uid) of EVERY requested memberId.
      const now = Date.now();
      const expiresAt = now + (30 * 24 * 60 * 60 * 1000);

      const results = [];
      const batch = admin.firestore().batch();

      for (const mid of memberIds) {
        const memberRef = admin.firestore().doc(`families/${familyCode}/members/${mid}`);
        const member = await memberRef.get();
        if (!member.exists) throw new HttpsError('not-found', `Member ${mid} not found`);
        const mdata = member.data();

        if (type === 'migration') {
          if (!isOwner) throw new HttpsError('permission-denied', 'Only owner can mint migration tokens');
          if (mdata.uid) throw new HttpsError('failed-precondition', `Member ${mid} already claimed`);
        } else {
          // graduation: requester must be managedBy on this member
          if (mdata.managedBy !== uid) throw new HttpsError('permission-denied', `You do not manage member ${mid}`);
          if (mdata.uid) throw new HttpsError('failed-precondition', `Member ${mid} already claimed`);
        }

        const token = crypto.randomBytes(32).toString('base64url');
        const tokenRef = admin.firestore().doc(`families/${familyCode}/claimTokens/${token}`);

        batch.set(tokenRef, {
          memberId: mid,
          type,
          createdBy: uid,
          createdAt: now,
          expiresAt,
          consumedAt: null,
          consumedBy: null,
          metadata: type === 'graduation' ? { graduation: true, wasManagedBy: uid } : { migration: true }
        });

        results.push({
          memberId: mid,
          memberName: mdata.name,
          token,
          deepLink: `https://couchtonight.app/?claim=${token}&family=${familyCode}`,
          expiresAt
        });
      }

      await batch.commit();
      return { ok: true, tokens: results };
    });
    ```

    **Step A — Add `isReadOnlyForMember` helper in js/app.js:**

    ```js
    function isReadOnlyForMember(member) {
      if (!member) return false;
      if (!state.settings || !state.settings.graceUntil) return false; // no settings loaded yet = be permissive
      if (Date.now() <= state.settings.graceUntil) return false; // still in grace
      // Post-grace:
      // - Authed members with uid: never read-only
      // - Sub-profiles (managedBy): never read-only on their own (they're act-as'd)
      // - Guests pre-expiry: never read-only (they have uid)
      // - Unclaimed adult members (no uid, no managedBy, not temporary): READ-ONLY
      if (member.uid) return false;
      if (member.managedBy) return false;
      if (member.temporary) return false;
      return true;
    }

    function isCurrentSelfReadOnly() {
      // When acting-as, use the sub-profile (never read-only). When self, use state.me.
      const effective = state.actingAs
        ? state.members.find(m => m.id === state.actingAs)
        : state.me;
      return isReadOnlyForMember(effective) || (!state.auth && !state.me);
    }
    ```
