    <!doctype html>
    <html lang="en">
    <head>
    <meta charset="utf-8">
    <meta name="viewport" content="width=device-width,initial-scale=1,viewport-fit=cover">

    <link rel="preconnect" href="https://fonts.gstatic.com" crossorigin>
    <link rel="stylesheet" href="https://fonts.googleapis.com/css2?family=Fraunces:wght@400&family=Inter:wght@400;500;600;700&family=Instrument+Serif:ital@0;1&display=swap">
    <link rel="stylesheet" href="/css/rsvp.css">
    </head>
    <body>
      <main class="rsvp-shell">
        <header class="rsvp-hero">
          <img class="rsvp-wordmark" src="/logo-h200.png" alt="Couch" width="423" height="200">
          <p class="rsvp-tagline"><em>You're invited.</em></p>
        </header>

        <div id="rsvp-content">
          <!-- Skeleton — populated by inline JS below -->

      <script>
      (function() {
        // Parse /rsvp/<token> from URL
        const path = window.location.pathname;
        const match = path.match(/^\/rsvp\/([^/?#]+)/);
        const token = match ? decodeURIComponent(match[1]) : null;
        const contentEl = document.getElementById('rsvp-content');

        const CF_BASE = 'https://us-central1-queuenight-84044.cloudfunctions.net';

        if (!token) {
          renderExpired();
          return;
        }

        // Hydrate event details (call rsvpInfo CF — read-only, returns {valid, expired, wp, hostName} or {valid:false})
        // For v1 simplicity: skip the lookup and let RSVP submit handle validation. Render a minimal generic invite card.
        renderInviteForm({ titleName: '', startTime: '', hostName: '', poster: '' });

        function renderInviteForm(invite) {
          contentEl.innerHTML = `
            <article class="rsvp-event-card">
              ${invite.poster ? `<div class="rsvp-event-poster" style="background-image:url('${escapeAttr(invite.poster)}')"></div>` : ''}
              <h1 class="rsvp-event-title">${esc(invite.titleName) || 'Couch night'}</h1>
              <div class="rsvp-event-meta">${esc(invite.startTime) || ''}</div>
              ${invite.hostName ? `<div class="rsvp-event-host"><em>${esc(invite.hostName)} invited you</em></div>` : ''}
            </article>

            <input class="rsvp-name-input" id="rsvp-name" type="text" placeholder="Your name" autocomplete="name" maxlength="40">

            <div id="rsvp-error" style="display:none;" class="rsvp-error"></div>

            <div class="rsvp-actions">
              <button class="rsvp-btn rsvp-yes" data-resp="yes" onclick="submitRsvp(this)">Going</button>
              <button class="rsvp-btn rsvp-maybe" data-resp="maybe" onclick="submitRsvp(this)">Maybe</button>
              <button class="rsvp-btn rsvp-no" data-resp="no" onclick="submitRsvp(this)">Can&rsquo;t make it</button>
            </div>
          `;
        }

        function renderConfirmation(resp, hostName) {
          const titles = { yes: "*See you there.*", maybe: "*Noted &mdash; we'll remind you.*", no: "*No worries.*" };
          const subs = {
            yes: hostName ? `Couch sent ${esc(hostName)} a heads up.` : `Couch sent the host a heads up.`,
            maybe: `Couch will nudge you closer to the time.`,
            no: `Next time.`
          };
          contentEl.innerHTML = `
            <div class="rsvp-confirmation">
              <div class="rsvp-confirmation-eyebrow">CONFIRMED</div>
              <div class="rsvp-confirmation-title">${titles[resp]}</div>
              <div class="rsvp-confirmation-sub">${subs[resp]}</div>
              <a href="/" class="rsvp-install-cta">Pull up a seat</a>
            </div>
          `;
        }

        function renderExpired() {
          contentEl.innerHTML = `
            <div class="rsvp-expired">
              <div class="rsvp-expired-title"><em>This invite has expired.</em></div>
              <div class="rsvp-expired-sub">Ask the host to send a fresh link.</div>
            </div>
          `;
        }

        function showError(msg) {
          const el = document.getElementById('rsvp-error');
          if (el) { el.textContent = msg; el.style.display = ''; }
        }

        window.submitRsvp = async function(btn) {
          const resp = btn.getAttribute('data-resp');
          const nameEl = document.getElementById('rsvp-name');
          const name = nameEl ? nameEl.value.trim() : '';
          if (!name && resp !== 'no') {
            showError("Add a name so the host knows who you are.");
            if (nameEl) nameEl.focus();
            return;
          }
          // Disable buttons during submit
          document.querySelectorAll('.rsvp-btn').forEach(b => b.disabled = true);
          btn.classList.add('tapped');
          try {
            const r = await fetch(CF_BASE + '/rsvpSubmit', {
              method: 'POST',
              headers: { 'Content-Type': 'application/json' },
              body: JSON.stringify({ data: { token: token, response: resp, name: name } })
            });
            const data = await r.json();
            if (data.error) {
              showError(data.error.message || "Couldn't save — try again in a moment.");
              document.querySelectorAll('.rsvp-btn').forEach(b => b.disabled = false);
              btn.classList.remove('tapped');
              return;
            }
            if (data.result && data.result.expired) {
              renderExpired();
              return;
            }
            const hostName = (data.result && data.result.hostName) || '';
            renderConfirmation(resp, hostName);
          } catch(e) {
            showError("Couldn't save — try again in a moment.");
            document.querySelectorAll('.rsvp-btn').forEach(b => b.disabled = false);
            btn.classList.remove('tapped');
          }
        };

        function esc(s) {
          if (s == null) return '';
          return String(s).replace(/[&<>"']/g, c => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'})[c]);
        }
        function escapeAttr(s) { return esc(s); }
      })();
      </script>
    </body>
    </html>

    // Phase 11 / REFR-05 — Unauth onCall to accept RSVP from /rsvp/<token> standalone page
    // Pattern: based on consumeGuestInvite.js (admin-SDK, idempotent, token-validated)

    const { onCall, HttpsError } = require('firebase-functions/v2/https');
    const admin = require('firebase-admin');
    const crypto = require('crypto');

    // Initialize admin if not already (matches existing index.js pattern)
    if (!admin.apps.length) admin.initializeApp();
    const db = admin.firestore();

    function guestKeyFor(wpId, name) {
      const h = crypto.createHash('sha256');
      h.update(`${wpId}-${(name || '').trim().toLowerCase()}`);
      return 'guest_' + h.digest('hex').slice(0, 16);
    }

    function looksLikeWpId(s) {
      return typeof s === 'string' && s.length >= 8 && s.length <= 64 && /^[A-Za-z0-9_-]+$/.test(s);
    }

    exports.rsvpSubmit = onCall({
      region: 'us-central1',
      cors: ['https://couchtonight.app', 'https://queuenight-84044.web.app'],
      memory: '256MiB',
      timeoutSeconds: 30
    }, async (request) => {
      const { token, response, name } = (request.data || {});

      // Validate inputs
      if (!looksLikeWpId(token)) {
        throw new HttpsError('invalid-argument', 'Invalid invite token.');
      }
      if (!['yes', 'maybe', 'no'].includes(response)) {
        throw new HttpsError('invalid-argument', 'Response must be yes, maybe, or no.');
      }
      const cleanName = (name || '').toString().trim().slice(0, 40);
      if (response !== 'no' && !cleanName) {
        throw new HttpsError('invalid-argument', 'Name is required for Going / Maybe.');
      }

      // Find the watchparty doc by id (token == wpId for v1)
      // Schema: families/{familyCode}/watchparties/{wpId}
      // Need to scan families for a matching wpId — or store a top-level rsvpTokens index.
      // For v1: scan families. Optimization: limit to recently-created families OR add a top-level token-to-family map.
      const familiesSnap = await db.collection('families').get();
      let wpRef = null;
      let wpDoc = null;
      let familyCode = null;
      for (const f of familiesSnap.docs) {
        const candidateRef = db.collection('families').doc(f.id).collection('watchparties').doc(token);
        const candidateDoc = await candidateRef.get();
        if (candidateDoc.exists) {
          wpRef = candidateRef;
          wpDoc = candidateDoc;
          familyCode = f.id;
          break;
        }
      }
      if (!wpRef || !wpDoc) {
        throw new HttpsError('not-found', 'This invite has expired or is invalid.');
      }

      const wp = wpDoc.data();
      const now = Date.now();
      // Expired: started >6h ago OR ended already
      if (wp.startAt && now > wp.startAt + 6 * 3600 * 1000) {
        return { expired: true };
      }
      if (wp.state === 'ended') {
        return { expired: true };
      }

      const guestKey = guestKeyFor(token, cleanName);
      const updates = {
        [`participants.${guestKey}`]: {
          name: cleanName || 'Guest',
          response: response,
          rsvpedAt: admin.firestore.FieldValue.serverTimestamp(),
          isGuest: true
        }
      };

      await wpRef.update(updates);

      let hostName = '';

| T-11-04-01 | Spoofing | Adversary submits arbitrary RSVPs to other users' watchparties using guessed/leaked tokens | mitigate | wpId token entropy: Firestore auto-generated IDs are 20-char base62 (~119 bits) — not guessable. Token leak requires the host already shared the link. CORS restricts callable to couchtonight.app + queuenight-84044.web.app ONLY (localhost origins intentionally OMITTED from production — local dev uses Firebase Functions emulator instead). Guest write is per-(wpId, name) idempotent — adversary can't flood unique entries. RECOMMEND adding rate limiting via Firebase App Check in Milestone 2. |
| T-11-04-02 | Tampering | Adversary modifies rsvp.html in browser to send `response: "admin"` or other invalid values | mitigate | rsvpSubmit validates `response in [yes,maybe,no]` server-side; rejects with HttpsError invalid-argument. Name capped at 40 chars + .trim(). Token validated via charset regex + Firestore lookup (not-found if no match). |
| T-11-04-03 | Repudiation | Guest claims they didn't RSVP / claims they RSVP'd Yes | mitigate | rsvpedAt server-side timestamp via FieldValue.serverTimestamp(). isGuest:true marks all unauth-submitted entries.
