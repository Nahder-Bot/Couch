    const SPORTS_PROVIDER_CONFIG = {
      // league key → { provider: 'espn'|'balldontlie', sport, league, balldontlieUrl }
      nfl: { provider: 'espn', espnSport: 'football', espnLeague: 'nfl' },
      nba: { provider: 'espn', espnSport: 'basketball', espnLeague: 'nba' },
      mlb: { provider: 'espn', espnSport: 'baseball', espnLeague: 'mlb' },
      nhl: { provider: 'espn', espnSport: 'hockey', espnLeague: 'nhl' },
      // BALLDONTLIE used as fallback if ESPN ever rate-limits / denies
    };

    const sportsScoreCache = {};   // key: gameId-bucket
    const sportsPlaysCache = {};
    const SCORE_CACHE_BUCKET_MS = 15 * 1000;
    const PLAYS_CACHE_BUCKET_MS = 30 * 1000;

    function bucketKey(gameId, bucketMs) {
      return `${gameId}-${Math.floor(Date.now() / bucketMs)}`;
    }

    const SportsDataProvider = {
      async getSchedule(leagueKey, daysAhead = 7) {
        // Reuses existing loadSportsGames pattern but returns parsed events
        const cfg = SPORTS_PROVIDER_CONFIG[leagueKey];
        if (!cfg) throw new Error('Unknown league: ' + leagueKey);
        if (cfg.provider === 'espn') {
          const allGames = [];
          const today = new Date();
          const fetches = [];
          for (let i = 0; i < daysAhead; i++) {
            const d = new Date(today); d.setDate(d.getDate() + i);
            const dateStr = d.getFullYear() + String(d.getMonth()+1).padStart(2,'0') + String(d.getDate()).padStart(2,'0');
            const url = `https://site.api.espn.com/apis/site/v2/sports/${cfg.espnSport}/${cfg.espnLeague}/scoreboard?dates=${dateStr}`;
            fetches.push(fetch(url).then(r => r.json()).catch(() => null));
          }
          const results = await Promise.all(fetches);
          results.forEach(data => {
            if (data && Array.isArray(data.events)) {
              data.events.forEach(ev => {
                const parsed = parseEspnEvent(ev);
                if (parsed) allGames.push(parsed);
              });
            }
          });
          return allGames;
        }
        // BALLDONTLIE branch (placeholder — implement when needed):
        // const url = `https://www.balldontlie.io/api/v1/games?seasons[]=2025&per_page=100&...`;
        return [];
      },

      async getScore(gameId, leagueKey) {
        const key = bucketKey(gameId, SCORE_CACHE_BUCKET_MS);
        if (sportsScoreCache[key]) return sportsScoreCache[key];
        const cfg = SPORTS_PROVIDER_CONFIG[leagueKey];
        if (!cfg || cfg.provider !== 'espn') return null;
        try {
          const url = `https://site.api.espn.com/apis/site/v2/sports/${cfg.espnSport}/${cfg.espnLeague}/scoreboard/${encodeURIComponent(gameId)}`;
          const r = await fetch(url);
          const d = await r.json();
          // ESPN returns event with competitions[0].competitors with .score field; status with type.state in {'pre','in','post'}
          const ev = d || {};
          const comp = (ev.competitions && ev.competitions[0]) || {};
          const teams = comp.competitors || [];
          const home = teams.find(t => t.homeAway === 'home') || {};
          const away = teams.find(t => t.homeAway === 'away') || {};
          const status = (ev.status && ev.status.type) || (comp.status && comp.status.type) || {};
          const score = {
            gameId,
            homeScore: parseInt(home.score, 10) || 0,
            awayScore: parseInt(away.score, 10) || 0,
            homeAbbr: (home.team && home.team.abbreviation) || '',
            awayAbbr: (away.team && away.team.abbreviation) || '',
            period: status.period || (ev.status && ev.status.period) || 0,
            clock: (ev.status && ev.status.displayClock) || '',
            state: status.state || 'unknown',  // pre / in / post
            statusDetail: status.shortDetail || '',
            ts: Date.now()
          };
          sportsScoreCache[key] = score;
          return score;
        } catch(e) {
          return null;
        }
      },

      async getPlays(gameId, leagueKey, sinceTs) {
        const key = bucketKey(gameId, PLAYS_CACHE_BUCKET_MS);
        if (sportsPlaysCache[key]) return sportsPlaysCache[key];
        const cfg = SPORTS_PROVIDER_CONFIG[leagueKey];
        if (!cfg || cfg.provider !== 'espn') return [];
        try {
          const url = `https://site.api.espn.com/apis/site/v2/sports/${cfg.espnSport}/${cfg.espnLeague}/summary?event=${encodeURIComponent(gameId)}`;
          const r = await fetch(url);
          const d = await r.json();
          // ESPN summary contains plays under d.plays or d.drives; pick last 10
          const plays = (d.plays || []).slice(-10).map(p => ({
            id: p.id || p.sequenceNumber,
            text: p.text || p.shortText || '',
            scoringPlay: !!p.scoringPlay,
            period: (p.period && p.period.number) || 0,
            clock: (p.clock && p.clock.displayValue) || '',
            ts: Date.now()
          }));
          sportsPlaysCache[key] = plays;
          return plays;
        } catch(e) {
          return [];
        }
      }
    };

    window.openGamePicker = function() {
      document.getElementById('game-picker-modal-bg').classList.add('on');

      document.getElementById('game-picker-modal-bg').classList.remove('on');
      _gamePickerSelected = null;
    };

    window.loadGamePickerLeague = async function(leagueKey) {
      _gamePickerLeague = leagueKey;
      // Highlight league button
      document.querySelectorAll('.game-picker-league-btn').forEach(b => b.classList.toggle('on', b.getAttribute('data-league') === leagueKey));
      const listEl = document.getElementById('game-picker-list');
      listEl.innerHTML = '<div class="game-picker-empty"><em>Loading games…</em></div>';
      try {
        const games = await SportsDataProvider.getSchedule(leagueKey, 7);
        if (!games.length) {
          listEl.innerHTML = '<div class="game-picker-empty"><em>No live games right now &mdash; check back near kickoff.</em></div>';
          return;
        }
        listEl.innerHTML = games.map(g => {
          const kickoff = g.startTime ? new Date(g.startTime).toLocaleString([], { weekday: 'short', month: 'short', day: 'numeric', hour: 'numeric', minute: '2-digit' }) : '';

            <div class="game-picker-at">@</div>

          </div>`;
        }).join('');
      } catch(e) {

    window.selectGame = function(gameId, gameJson) {

      const confirmBtn = document.getElementById('game-picker-confirm');
      if (confirmBtn) confirmBtn.disabled = false;

      if (!_gamePickerSelected || !state.me) return;

        hostId: state.me.id,

          league: _gamePickerSelected.league,
          homeTeam: _gamePickerSelected.homeTeam,
          awayTeam: _gamePickerSelected.awayTeam,
          leagueEmoji: { nfl: '🏈', nba: '🏀', mlb: '⚾', nhl: '🏒' }[_gamePickerSelected.league] || '🎮'
        },

        }
      } catch(e) {
