    // ---- Phase 11 / REFR-13 — Couch Nights themed packs ----

    let _activePackId = null;

    function renderCouchNightsRow() {
      const row = document.getElementById('couch-nights-row');
      if (!row || typeof COUCH_NIGHTS_PACKS === 'undefined') return;
      if (!COUCH_NIGHTS_PACKS.length) { row.innerHTML = ''; return; }
      row.innerHTML = COUCH_NIGHTS_PACKS.map(pack => `
        <div class="couch-night-tile" onclick="openCouchNightPack('${pack.id}')" style="background-image:linear-gradient(0deg,rgba(20,17,15,0.85),transparent 60%),url('${escapeAttr(pack.heroImageUrl)}')">
          <div class="couch-night-tile-title">${escapeHtml(pack.title)}</div>
        </div>
      `).join('');
    }

    window.openCouchNightPack = function(packId) {
      const pack = COUCH_NIGHTS_PACKS.find(p => p.id === packId);
      if (!pack) return;
      _activePackId = packId;
      document.getElementById('couch-night-hero').style.backgroundImage = `url('${pack.heroImageUrl}')`;
      document.getElementById('couch-night-sheet-title').textContent = pack.title;
      document.getElementById('couch-night-sheet-desc').innerHTML = `<em>${pack.description}</em>`;
      const previewEl = document.getElementById('couch-night-titles-preview');
      previewEl.innerHTML = '<div class="sk-row-posters"><div class="sk-poster"></div><div class="sk-poster"></div><div class="sk-poster"></div></div>';
      document.getElementById('couch-night-sheet-bg').classList.add('on');
      loadPackPreview(pack);
    };

    window.closeCouchNightSheet = function() {
      document.getElementById('couch-night-sheet-bg').classList.remove('on');
      _activePackId = null;
    };

    async function loadPackPreview(pack) {
      const previewEl = document.getElementById('couch-night-titles-preview');
      try {
        const ids = pack.tmdbIds.slice(0, 8);
        const fetches = ids.map(id => tmdbFetch(`/movie/${id}`).catch(() => null));
        const results = await Promise.all(fetches);
        const valid = results.filter(r => r && r.poster_path);
        if (!valid.length) {
          previewEl.innerHTML = '<div class="discover-loading"><em>This pack\'s being restocked. Try another.</em></div>';
          return;
        }
        previewEl.innerHTML = valid.map(t => `
          <div class="couch-night-preview-card">
            <div class="couch-night-preview-poster" style="background-image:url('https://image.tmdb.org/t/p/w185${t.poster_path}')"></div>
            <div class="couch-night-preview-title">${escapeHtml(t.title || t.name || '')}</div>
          </div>
        `).join('');
      } catch(e) {
        previewEl.innerHTML = '<div class="discover-loading"><em>Could not load pack.</em></div>';
      }
    }

    window.confirmStartPack = async function() {
      if (!_activePackId || !state.me) return;
      if (guardReadOnlyWrite && guardReadOnlyWrite()) return;
      const pack = COUCH_NIGHTS_PACKS.find(p => p.id === _activePackId);
      if (!pack) return;
      const btn = document.getElementById('start-pack-cta');
      btn.disabled = true;
      btn.textContent = 'Seeding…';
      try {
        await seedBallotFromPack(pack);
        haptic('success');
        flashToast(`Pack loaded: ${pack.title}`, { kind: 'success' });
        closeCouchNightSheet();
        // Launch Vote mode
        if (typeof openSwipeMode === 'function') openSwipeMode();
        else if (typeof window.openSwipeMode === 'function') window.openSwipeMode();
      } catch(e) {
        flashToast('Could not load pack', { kind: 'error' });
        btn.disabled = false;
        btn.textContent = 'Start this pack';
      }
    };

    async function seedBallotFromPack(pack) {
      // For each TMDB ID: fetch metadata → add to state.titles via existing addTitle flow
      // Strategy: reuse existing addTitleToCouch / addFromAddTab pattern.
      // Grep for the existing "add a title to the family couch list" function name.
      // Candidates: addTitleToCouch, addTitleFromTmdb, addFromAddTab, addTitle — whichever exists.
      for (const id of pack.tmdbIds) {
        try {
          // Option 1 — reuse existing function (preferred):
          if (typeof addTitleFromTmdb === 'function') {
            await addTitleFromTmdb(id, { source: `pack:${pack.id}` });
            continue;
          }
          if (typeof addFromAddTab === 'function') {
            // addFromAddTab takes (rowId, tmdbId) — adapt as needed
            await addFromAddTab(`pack-${pack.id}`, id);
            continue;
          }
          // Option 2 — fallback inline: fetch TMDB → addDoc to family titles collection
          const d = await tmdbFetch(`/movie/${id}`);
          if (!d || !d.poster_path) continue;
          await addDoc(titlesRef, {
            tmdbId: id,
            name: d.title || d.name,
            year: (d.release_date || d.first_air_date || '').slice(0, 4),
            kind: 'movie',
            posterPath: d.poster_path,
            poster: `https://image.tmdb.org/t/p/w300${d.poster_path}`,
            genres: (d.genres || []).map(g => g.id),
            runtime: d.runtime,
            overview: d.overview,
            addedVia: `pack:${pack.id}`,
            addedAt: Date.now(),
            addedBy: state.me.id,
            ...writeAttribution()
          });
        } catch(e) { /* skip failed fetch */ }
      }
    }
