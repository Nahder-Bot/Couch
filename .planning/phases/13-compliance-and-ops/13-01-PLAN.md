```javascript
exports.requestAccountDeletion = onCall({
  region: 'us-central1',
  cors: ['https://couchtonight.app', 'https://queuenight-84044.web.app'],
  memory: '256MiB',
  timeoutSeconds: 30
}, async (req) => {
  // T-1 mitigation: derive uid from auth context, NEVER from request payload
  const uid = req.auth && req.auth.uid;
  if (!uid) throw new HttpsError('unauthenticated', 'Sign in required');

  // Confirmation gate (mirrors rsvpSubmit input-validation pattern)
  const { confirm } = (req.data || {});
  if (confirm !== 'DELETE') {
    throw new HttpsError('failed-precondition', 'Confirmation phrase missing or wrong.');
  }

  // Pitfall 1 mitigation: owner-of-family pre-flight inside the callable too
  // (defense-in-depth — UI also calls checkAccountDeleteEligibility first, but
  // a malicious client could skip the pre-flight)
  const owned = await db.collection('families').where('ownerUid', '==', uid).get();
  if (!owned.empty) {
    throw new HttpsError('failed-precondition',
      `Transfer ownership of ${owned.size} family/families before deleting your account.`);
  }

  // Soft-delete: set 14-day timer (per CONTEXT.md addendum #2)
  const FOURTEEN_DAYS_MS = 14 * 24 * 60 * 60 * 1000;
  const hardDeleteAt = Date.now() + FOURTEEN_DAYS_MS;

  await db.collection('users').doc(uid).set({
    deletionRequestedAt: admin.firestore.FieldValue.serverTimestamp(),
    hardDeleteAt: hardDeleteAt
  }, { merge: true });

  return { success: true, hardDeleteAt };
});
```

**Step 1.4 — Create `C:\Users\nahde\queuenight\functions\src\accountDeletionReaper.js` (NEW).**

This is the largest file — full cascading delete. Skeleton from PATTERNS.md "deletionReaperTick.js" + RESEARCH.md "Pattern 2" + "Storage path enumeration":

```javascript
'use strict';
const { onSchedule } = require('firebase-functions/v2/scheduler');
const admin = require('firebase-admin');
const firebaseTools = require('firebase-tools');

if (!admin.apps.length) admin.initializeApp();
const db = admin.firestore();

exports.accountDeletionReaper = onSchedule({
  schedule: 'every 1 hours',
  region: 'us-central1',
  timeoutSeconds: 540,
  memory: '512MiB'
}, async (event) => {
  const now = Date.now();
  let processed = 0, errored = 0;

  // Find users whose grace window has expired
  const due = await db.collection('users')
    .where('hardDeleteAt', '<=', now)
    .limit(20)
    .get();

  for (const userDoc of due.docs) {
    const uid = userDoc.id;
    const sweepCounts = { members: 0, vetoes: 0, comments: 0, activity: 0, lists: 0, intents: 0, watchparties: 0, photos: 0, sessions: 0 };

    try {
      // T-2 mitigation: re-check deletionRequestedAt + hardDeleteAt INSIDE
      // a transaction-like read at hard-delete time. If user signed back in
      // and called cancelAccountDeletion, hardDeleteAt would have been deleted
      // and they wouldn't appear in the .where('hardDeleteAt', '<=', now) query
      // — but we re-read for defense-in-depth.
      const fresh = await db.collection('users').doc(uid).get();
      const data = fresh.exists ? fresh.data() : null;
      if (!data || !data.hardDeleteAt || data.hardDeleteAt > now) {
        console.log('accountDeletionReaper skip uid (no longer due)', uid);
        continue;
      }

      // 1. Find all families this user belongs to (via reverse-index subcollection)
      const familiesSnap = await db.collection('users').doc(uid).collection('groups').get();
      const familyCodes = familiesSnap.docs.map(d => d.id);

      // Pitfall 1 / T-2 second pass: also check ownerUid (in case ownership changed
      // mid-grace; if user re-claimed ownership somehow, abort this user's hard-delete).
      const ownedNow = await db.collection('families').where('ownerUid', '==', uid).get();
      if (!ownedNow.empty) {
        console.warn('accountDeletionReaper aborting uid (now owns families)', uid, ownedNow.size);
        await db.collection('users').doc(uid).set({
          lastDeletionAttemptError: 'User owns families at hard-delete time; aborting.',
          lastDeletionAttemptAt: admin.firestore.FieldValue.serverTimestamp()
        }, { merge: true });
        errored++;
        continue;
      }

      // 2. Per-family sweeps (all 12 paths from interfaces block)
      for (const familyCode of familyCodes) {
        const famRef = db.collection('families').doc(familyCode);
        const famDoc = await famRef.get();
        if (!famDoc.exists) continue;

        // 2a. members where uid==X
        const myMembers = await famRef.collection('members').where('uid', '==', uid).get();
        for (const m of myMembers.docs) { await m.ref.delete(); sweepCounts.members++; }

        // 2b. vetoHistory where actingUid==X
        const myVetoes = await famRef.collection('vetoHistory').where('actingUid', '==', uid).get();
        for (const v of myVetoes.docs) { await v.ref.delete(); sweepCounts.vetoes++; }

        // 2c. activity where actingUid==X
        const myActivity = await famRef.collection('activity').where('actingUid', '==', uid).get();
        for (const a of myActivity.docs) { await a.ref.delete(); sweepCounts.activity++; }

        // 2d. lists where actingUid==X
        const myLists = await famRef.collection('lists').where('actingUid', '==', uid).get();
        for (const l of myLists.docs) { await l.ref.delete(); sweepCounts.lists++; }

        // 2e. intents — delete if creator, else strip from rsvps map
        const allIntents = await famRef.collection('intents').get();
        for (const intentDoc of allIntents.docs) {
          const id = intentDoc.data() || {};
          if (id.createdByUid === uid) {
            await intentDoc.ref.delete();
            sweepCounts.intents++;
          } else if (id.rsvps && id.rsvps[uid]) {
            await intentDoc.ref.update({ [`rsvps.${uid}`]: admin.firestore.FieldValue.delete() });
            sweepCounts.intents++;
          }
        }

        // 2f. watchparties — strip from participants; if hostUid && active, end first
        const allWp = await famRef.collection('watchparties').get();
        for (const wpDoc of allWp.docs) {
          const wp = wpDoc.data() || {};
          if (wp.hostUid === uid && wp.status && wp.status !== 'ended' && wp.status !== 'cancelled') {
            await wpDoc.ref.update({ status: 'ended', endedAt: admin.firestore.FieldValue.serverTimestamp(), endedReason: 'host-account-deleted' });
          }
          if (wp.participants && wp.participants[uid]) {
            await wpDoc.ref.update({ [`participants.${uid}`]: admin.firestore.FieldValue.delete() });
            sweepCounts.watchparties++;
          }
        }

        // 2g. titles.userMoods.{uid} — strip; do NOT delete title docs (family-shared)
        // Titles can be many; this is a targeted update
        const titlesWithMyMood = await famRef.collection('titles').where(`userMoods.${uid}`, '!=', null).get().catch(() => null);
        if (titlesWithMyMood && titlesWithMyMood.docs) {
          for (const t of titlesWithMyMood.docs) {
            await t.ref.update({ [`userMoods.${uid}`]: admin.firestore.FieldValue.delete() });
          }
        }

        // 2h. comments where actingUid==X — collectionGroup query
        // (comments live at families/{code}/titles/{titleId}/comments/{cid};
        //  scope by family path prefix for safety)
        // Use collectionGroup to avoid iterating every title
        const myCommentsCG = await db.collectionGroup('comments').where('actingUid', '==', uid).get();
        for (const c of myCommentsCG.docs) {
          // verify path is under our families/{familyCode} prefix
          if (c.ref.path.startsWith(`families/${familyCode}/`)) {
            await c.ref.delete();
            sweepCounts.comments++;
          }
        }

        // 2i. sessions vote/veto attribution — best-effort; sessions hold daily state
        // Skip deep mutation here; sessions are short-lived day-bucketed and roll over.
        // Document this in 13-01-SUMMARY.md under "Known limitations".
      }

      // 3. T-3 mitigation: Storage sweep at hard-delete time (NOT at soft-delete)
      // so photos uploaded during 14-day grace window are also swept.
      const bucket = admin.storage().bucket();
      for (const familyCode of familyCodes) {
        const [files] = await bucket.getFiles({ prefix: `couch-albums/${familyCode}/` });
        const mine = files.filter(f => f.name.includes(`_${uid}.jpg`));
        await Promise.all(mine.map(f => f.delete()));
        sweepCounts.photos += mine.length;
      }

      // 4. Recursive delete users/{uid} subtree (groups/{familyCode} lives there)
      await firebaseTools.firestore.delete(`users/${uid}`, {
```
