  region: 'us-central1',
  cors: ['https://couchtonight.app', 'https://queuenight-84044.web.app'],
  memory: '256MiB',
  timeoutSeconds: 30
}, async (req) => {
  // T-1 mitigation: derive uid from auth context, NEVER from request payload
  const uid = req.auth && req.auth.uid;
  if (!uid) throw new HttpsError('unauthenticated', 'Sign in required');

  // Confirmation gate (mirrors rsvpSubmit input-validation pattern)
  const { confirm } = (req.data || {});
  if (confirm !== 'DELETE') {
    throw new HttpsError('failed-precondition', 'Confirmation phrase missing or wrong.');

  // Pitfall 1 mitigation: owner-of-family pre-flight inside the callable too
  // (defense-in-depth — UI also calls checkAccountDeleteEligibility first, but
  // a malicious client could skip the pre-flight)
  const owned = await db.collection('families').where('ownerUid', '==', uid).get();

  const FOURTEEN_DAYS_MS = 14 * 24 * 60 * 60 * 1000;
  const hardDeleteAt = Date.now() + FOURTEEN_DAYS_MS;

  await db.collection('users').doc(uid).set({
    deletionRequestedAt: admin.firestore.FieldValue.serverTimestamp(),
    hardDeleteAt: hardDeleteAt
  }, { merge: true });

  return { success: true, hardDeleteAt };
});

async function recursiveDeleteRef(ref) {
  if (typeof db.recursiveDelete === 'function') {
    await db.recursiveDelete(ref);
    return;
  }
  // Fallback: BulkWriter-based recursion (Admin SDK >= 9 supports BulkWriter)
  const bulkWriter = db.bulkWriter();
  // Some Admin SDK versions support db.recursiveDelete(ref, bulkWriter);
  // others require manual traversal. Detect and pick the right path.
  if (typeof db.recursiveDelete === 'function') {

  } else {
    // Manual recursion as last resort
    await recursiveDeleteManual(ref, bulkWriter);
  }
  await bulkWriter.close();
}

async function recursiveDeleteManual(ref, bulkWriter) {
  // Walk subcollections and delete recursively. ref is a DocumentReference.
  const subcollections = await ref.listCollections();
  for (const sub of subcollections) {
    const docs = await sub.listDocuments();
    for (const d of docs) {
      await recursiveDeleteManual(d, bulkWriter);
    }
  }
  bulkWriter.delete(ref);
}

// Review fix HIGH-1: sessions sweep. Per CONTEXT.md / sessions-doc shape, families/{familyCode}/sessions/{dateKey}
// hold daily voting state. Older sessions are expirable (delete entirely); active sessions get scrubbed in place.
async function sweepSessionsForFamily(famRef, uid) {
  let deleted = 0, scrubbed = 0;
  const THIRTY_DAYS_MS = 30 * 24 * 60 * 60 * 1000;
  const now = Date.now();
  const todayKey = new Date().toISOString().slice(0, 10); // YYYY-MM-DD

  const sessionsSnap = await famRef.collection('sessions').get();
  for (const sessDoc of sessionsSnap.docs) {
    const dateKey = sessDoc.id; // typically "YYYY-MM-DD"
    const data = sessDoc.data() || {};
    // Determine if expired: dateKey is before today OR sessDoc was last touched > 30 days ago.
    // Fall back to age-only check if dateKey is unparseable.
    let isExpired = false;
    if (typeof dateKey === 'string' && /^\d{4}-\d{2}-\d{2}$/.test(dateKey)) {
      isExpired = (dateKey < todayKey);
    }
    if (!isExpired && data.lastUpdatedAt && typeof data.lastUpdatedAt.toMillis === 'function') {
      isExpired = (now - data.lastUpdatedAt.toMillis() > THIRTY_DAYS_MS);
    }

    if (isExpired) {
      // Past natural expiration — delete entirely.
      await sessDoc.ref.delete();
      deleted++;
      continue;
    }

    // Active session — scrub in place.
    // Build a single update payload covering all uid-attributable fields.
    const update = {};
    if (data.actingUid === uid) update.actingUid = null;
    if (data.participants && data.participants[uid]) {
      update[`participants.${uid}`] = admin.firestore.FieldValue.delete();
    }
    if (data.votes && typeof data.votes === 'object') {
      for (const titleId of Object.keys(data.votes)) {
        if (data.votes[titleId] && data.votes[titleId][uid]) {
          update[`votes.${titleId}.${uid}`] = admin.firestore.FieldValue.delete();
        }
      }
    }
    // Array filters: vetoes + chooserHistory drop entries whose uid matches.
    if (Array.isArray(data.vetoes)) {
      const filteredVetoes = data.vetoes.filter(v => !v || v.uid !== uid);
      if (filteredVetoes.length !== data.vetoes.length) update.vetoes = filteredVetoes;
    }
    if (Array.isArray(data.chooserHistory)) {
      const filteredChooser = data.chooserHistory.filter(c => !c || c.uid !== uid);
      if (filteredChooser.length !== data.chooserHistory.length) update.chooserHistory = filteredChooser;
    }
    if (Object.keys(update).length > 0) {
      await sessDoc.ref.update(update);
      scrubbed++;
    }
  }
  return { deleted, scrubbed };
}
