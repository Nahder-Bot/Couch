// === Phase 15 / S2 + S3 + S6 (TRACK-15-04..06) — Your couch's progress (tuple-aware) ===
// Detail-modal "YOUR COUCH'S PROGRESS" section. Sibling primitive to
// renderTvProgressSection (per-INDIVIDUAL, line 8291) — both coexist during v1
// per RESEARCH §Foundational Finding A5. Reads t.tupleProgress (15-02 schema).
// Renders rows sorted watchedAt desc, max 4; appends S6 mute toggle at bottom.
// Returns '' (hides section) when no TV / no tuples / title watched.
function renderCv15TupleProgressSection(t) {
  if (!t || t.kind !== 'TV' || t.watched) return '';
  const tuples = (t.tupleProgress && typeof t.tupleProgress === 'object') ? t.tupleProgress : {};
  const tupleKeys = Object.keys(tuples);
  if (tupleKeys.length === 0) return '';
  const meId = state.me && state.me.id;
  // Sort by watchedAt desc, slice max 4 (UI-SPEC §Cross-tuple visual handling).
  const sorted = tupleKeys
    .map(tk => ({ tk, prog: tuples[tk] || {} }))
    .sort((a, b) => (b.prog.updatedAt || 0) - (a.prog.updatedAt || 0));
  const visible = sorted.slice(0, 4);
  const overflow = sorted.length > 4 ? sorted.length - 4 : 0;
  const rows = visible.map(({ tk, prog }) => {
    const memberIds = tk.split(',').filter(Boolean);
    const named = (state.family && state.family.tupleNames && state.family.tupleNames[tk]) || null;
    const displayName = (named && named.name) || tupleDisplayName(tk, state.members);
    const isUnnamed = !displayName;
    const escId = escapeHtml(t.id);
    const escTk = escapeHtml(tk);
    const nameSpan = isUnnamed
      ? `<span class="cv15-tuple-name unnamed" id="cv15-tname-${escId}-${escTk}"><em>name this couch</em></span>`
      : `<span class="cv15-tuple-name" id="cv15-tname-${escId}-${escTk}">${escapeHtml(displayName)}</span>`;
    const renameBtn = `<button class="cv15-tuple-rename" type="button" aria-label="Rename this couch" onclick="cv15ShowRenameInput('${escId}','${escTk}', event)">&#9998;</button>`;
    const seasonNum = (prog.season != null) ? prog.season : '?';
    const episodeNum = (prog.episode != null) ? prog.episode : '?';
    const ago = prog.updatedAt ? cv15RelativeTime(prog.updatedAt) : '';
    return `<div class="cv15-progress-row" data-tk="${escTk}">
      <div class="cv15-progress-row-body">
        ${nameSpan}${renameBtn}
        <div class="cv15-progress-time">${escapeHtml(ago)}</div>
      </div>
      <div class="cv15-progress-row-actions">
        <div class="cv15-progress-pos">S${escapeHtml(String(seasonNum))} &middot; E${escapeHtml(String(episodeNum))}</div>
      </div>
    </div>`;
  }).join('');
  const overflowHtml = overflow > 0
    ? `<button class="cv15-mute-toggle" type="button" style="border-top:0;color:var(--ink-dim);" onclick="cv15ShowAllTuples('${escapeHtml(t.id)}')">View all (${sorted.length})</button>`
    : '';
  return `<div class="detail-section detail-cv15-progress">
    <h4>YOUR COUCH'S PROGRESS</h4>
    ${rows}
    ${overflowHtml}
    ${renderCv15MutedShowToggle(t)}
  </div>`;
}

// S6 per-show kill-switch text-link. Reads t.mutedShows[me.id].
function renderCv15MutedShowToggle(t) {
  if (!t) return '';
  const me = state.me;
  if (!me) return '';
  const muted = !!(t.mutedShows && t.mutedShows[me.id]);
  const label = muted ? 'Notifications off &middot; Re-enable' : 'Stop notifying me about this show';
  const cls = muted ? 'cv15-mute-toggle on' : 'cv15-mute-toggle';
  return `<button class="${cls}" type="button" onclick="toggleMutedShow('${escapeHtml(t.id)}')">${label}</button>`;
}

// Relative-time helper (avoids reading large date libs; mirror tone of
// existing time helpers like the watchparty banner). Sentence-case fragments.
function cv15RelativeTime(ts) {
  if (!ts) return '';
  const diff = Date.now() - ts;
  if (diff < 0) return 'just now';
  const mins = Math.floor(diff / 60000);
  if (mins < 1) return 'just now';
  if (mins < 60) return `${mins} min ago`;
  const hrs = Math.floor(mins / 60);
  if (hrs < 24) return `${hrs}h ago`;
  const days = Math.floor(hrs / 24);
  if (days === 1) return 'yesterday';
  if (days < 7) return `${days} days ago`;
  const weeks = Math.floor(days / 7);
  if (weeks < 5) return `${weeks}w ago`;
  const months = Math.floor(days / 30);
  return `${months}mo ago`;
}

function cv15RenameKeydown(ev) {

function cv15RenameBlur(ev) {

  if (ev.target.dataset.cancelled === '1') return;
  cv15SaveRenameInput(ev.target);
}
async function cv15SaveRenameInput(input) {
  if (!input) return;
  const tk = input.dataset.tk;
  const titleId = input.dataset.titleId;
  const value = (input.value || '').trim();
  // Detach handlers so we don't re-fire on re-render.
  input.removeEventListener('blur', cv15RenameBlur);
  input.removeEventListener('keydown', cv15RenameKeydown);
  await setTupleName(tk, value);
  flashToast('Saved');
  // The title-doc onSnapshot won't re-fire (we wrote families/* — different
  // collection). Trigger a manual re-render of the detail modal for THIS title
  // to pick up the new tupleNames map. Reads from state.family.tupleNames
  // which 15-02's snapshot extension already hydrated.
  if (typeof renderDetailShell === 'function' && state.titles) {
    const t = state.titles.find(x => x.id === titleId);
    if (t && typeof window.openDetailModal === 'function' && document.getElementById('detail-modal-bg')) {
      // Re-render only if modal is currently open (avoid unwanted side-effects).
      const dm = document.getElementById('detail-modal-bg');
      if (dm && dm.classList.contains('on')) {
        const body = document.getElementById('detail-modal-body');
        if (body) body.innerHTML = renderDetailShell(t);
      }
    }
  }
}
function cv15CancelRenameInput(input) {
  if (!input) return;
  input.dataset.cancelled = '1';
  // Re-render the detail modal to restore the original span.
  const titleId = input.dataset.titleId;
  if (state.titles) {
    const t = state.titles.find(x => x.id === titleId);
    if (t) {
      const body = document.getElementById('detail-modal-body');
      if (body && typeof renderDetailShell === 'function') body.innerHTML = renderDetailShell(t);
    }
  }
}
