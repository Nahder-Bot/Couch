```javascript
        // === Phase 15 / D-11 + D-13 + D-14 + D-15 + D-16 (TRACK-15-08) — live-release sweep ===
        // Per family, walk every TV title with non-empty t.tupleProgress. For each
        // tracked show, build the subscriber set (D-09/D-10 union-of-tuples MINUS
        // D-12 per-show muted), check the D-13 24h-before window using
        // t.nextEpisode.airDate (TMDB date-only — see RESEARCH §Q6), gate on
        // D-14 ≥2-trackers threshold, run D-16 watchparty suppression query
        // (REQUIRES the composite index from 15-01), and fire newSeasonAirDate
        // push via sendToMembers (gated by NOTIFICATION_DEFAULTS extended above).
        // Idempotency: t.liveReleaseFiredFor[`s{N}e{M}`] flag prevents re-fire.
        let titlesSnap;
        try {
          titlesSnap = await db.collection('families').doc(familyDoc.id).collection('titles').get();
        } catch (e) {
          console.warn('[15-06] titles list failed', familyDoc.id, e.message);
          continue;
        }
        for (const tdoc of titlesSnap.docs) {
          const t = tdoc.data();
          // Early exits — cheapest first.
          if (t.kind !== 'TV') continue;
          if (!t.tupleProgress || Object.keys(t.tupleProgress).length === 0) continue;
          const next = t.nextEpisode;
          if (!next || !next.airDate || next.season == null || next.episode == null) continue;

          // === D-09/D-10 — subscriber set: union-of-tuples ===
          const subscriberIds = new Set();
          for (const tk of Object.keys(t.tupleProgress)) {
            for (const mid of tk.split(',')) {
              if (mid) subscriberIds.add(mid);
            }
          }
          // === D-12 — per-show muted-by-member kill switch ===
          if (t.mutedShows) {
            for (const mid of Array.from(subscriberIds)) {
              if (t.mutedShows[mid]) subscriberIds.delete(mid);
            }
          }
          // === D-14 — ≥2 family members tracking the show ===
          if (subscriberIds.size < 2) continue;

          // === D-13 — 24h-before window (T-25h to T-23h) ===
          // TMDB exposes air_date as DATE-ONLY (RESEARCH §Q6). 9pm best-guess
          // for window math; user-facing copy uses {day}-only (RESEARCH §Q6
          // accepted v1 fallback).
          const airTs = new Date(next.airDate + 'T21:00:00').getTime();
          if (!Number.isFinite(airTs)) continue;
          const minsToAir = (airTs - now) / 60000;
          if (minsToAir <= 60 * 23 || minsToAir > 60 * 25) continue;

          // === Idempotency — already fired (or suppressed) ===
          const epKey = `s${next.season}e${next.episode}`;
          if (t.liveReleaseFiredFor && t.liveReleaseFiredFor[epKey]) continue;

          // === D-16 — already-scheduled wins ===
          // Suppression query: any non-archived, non-cancelled watchparty for
          // this title within ±90min of airTs. REQUIRES composite index on
          // (titleId asc, startAt asc) shipped in 15-01.
          let blocking = false;
          try {
            const wpSnap = await db.collection('families').doc(familyDoc.id)
              .collection('watchparties')
              .where('titleId', '==', tdoc.id)
              .where('startAt', '>=', airTs - 90 * 60000)
              .where('startAt', '<=', airTs + 90 * 60000)
              .get();
            for (const wpDoc of wpSnap.docs) {
              const wp = wpDoc.data();
              if (wp.status === 'archived' || wp.status === 'cancelled') continue;
              blocking = true;
              break;
            }
          } catch (e) {
            console.warn('[15-06] suppression query failed (index missing?)', familyDoc.id, tdoc.id, e.message);
            // Mark as fired anyway so we don't retry every tick on a known failure.
            try {
              await tdoc.ref.update({ [`liveReleaseFiredFor.${epKey}`]: 'suppression_query_failed' });
            } catch (_) {}
            continue;
          }
          if (blocking) {
            // Mark as suppressed (don't re-evaluate every 5min for this episode).
            try {
              await tdoc.ref.update({ [`liveReleaseFiredFor.${epKey}`]: 'suppressed_existing_wp' });
            } catch (e) { console.warn('[15-06] suppress flag write failed', e.message); }
            continue;
          }

          // === Fire the push ===
          // Day-name rendering — RESEARCH §Q6 + UI-SPEC §Copywriting Contract.
          // Coarse: "tonight" / "tomorrow" / "{Weekday}".
          const daysOut = Math.round((airTs - now) / 86400000);
          const when = daysOut === 0 ? 'tonight'
                     : daysOut === 1 ? 'tomorrow'
                     : new Date(airTs).toLocaleDateString('en-US', { weekday: 'long' });
          const showName = t.name || 'A show you watch';
          const body = `${showName} S${next.season}E${next.episode} airs ${when}. Watch with the couch?`;

          // Set the idempotency flag FIRST (per Phase 14 idempotency-stamp pattern at
          // ~functions/index.js:792-803 — flag-then-fire). If push fails after flag
          // is set we lose the push for this episode — acceptable per D-15 (every
          // episode is a fresh prompt; missing one is bounded).
          try {
            await tdoc.ref.update({ [`liveReleaseFiredFor.${epKey}`]: now });
          } catch (e) {
            console.warn('[15-06] fire flag write failed; skipping push', e.message);
            continue;
          }
          try {
            await sendToMembers(familyDoc.id, Array.from(subscriberIds), {
              title: 'Live tonight',
              body: body,
              tag: `live-release-${tdoc.id}-${epKey}`,
              url: `/?nominate=${tdoc.id}&prefillTime=${airTs}`
            }, { eventType: 'newSeasonAirDate' });
          } catch (e) {
            console.warn('[15-06] live-release push failed', familyDoc.id, tdoc.id, epKey, e.message);
            // Flag stays set — accept push loss per D-15 cadence model.
          }
        }
        // === /Phase 15 sweep ===
```
