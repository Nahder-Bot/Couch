        // Per family, walk every TV title with non-empty t.tupleProgress.
        // For each tracked show, build the subscriber set (D-09/D-10 union-of-tuples
        // MINUS D-12 per-show muted), check D-13 24h-before window using
        // t.nextEpisode.airDate, gate on D-14 ≥2-trackers threshold, run D-16
        // watchparty suppression query, and fire newSeasonAirDate push.

        // REVIEW HIGH-3: this loop runs ONCE PER HOUR per family, not every
        // 5-min tick. Throttle cursor: families/{code}.liveReleaseSweepLastRunAt.
        // REVIEW HIGH-4: titles with stale t.nextEpisodeRefreshedAt (>7 days)
        // are SKIPPED with Sentry breadcrumb so we can observe gap frequency.
        // REVIEW LOW/MEDIUM-11: window upper bound widened to 26h to absorb
        // tick-drift (now that we run hourly, not every 5min).
        // REVIEW MEDIUM-12: push body uses "New episode tonight/tomorrow/{Weekday}"
        // framing aligned with per-episode semantic (not new-season-only).
        const HOURLY_MS = 60 * 60 * 1000;
        const STALE_NEXT_EPISODE_MS = 7 * 24 * 60 * 60 * 1000;  // 7 days
        const familyData = familyDoc.data();
        const lastSweep = familyData.liveReleaseSweepLastRunAt || 0;
        if ((now - lastSweep) < HOURLY_MS) {
          // REVIEW HIGH-3 — too soon since last sweep; skip until next hour boundary.
          continue;
        }

        let titlesSnap;
        try {
          titlesSnap = await db.collection('families').doc(familyDoc.id).collection('titles').get();
        } catch (e) {
          console.warn('[15-06] titles list failed', familyDoc.id, e.message);
          continue;
        }

        let sweptCount = 0;
        let staleCount = 0;
        let firedCount = 0;
        let suppressedCount = 0;

        for (const tdoc of titlesSnap.docs) {
          const t = tdoc.data();
          // Cheapest early exits first.
          if (t.kind !== 'TV') continue;
          if (!t.tupleProgress || Object.keys(t.tupleProgress).length === 0) continue;

          // === REVIEW HIGH-4 — stale-data guard ===
          // If nextEpisode missing OR nextEpisodeRefreshedAt > 7 days old, skip.
          const refreshedAt = t.nextEpisodeRefreshedAt || 0;
          const isStale = !t.nextEpisode || (now - refreshedAt) > STALE_NEXT_EPISODE_MS;
          if (isStale) {
            staleCount++;
            // Stamp once per title to keep telemetry actionable (not per-tick noise).
            const lastStaleStamp = t.liveReleaseStaleLoggedAt || 0;
            if ((now - lastStaleStamp) > 24 * 60 * 60 * 1000) {  // log at most daily
              try {
                await tdoc.ref.update({ liveReleaseStaleLoggedAt: now });
              } catch (_) {}
              try {
                if (typeof Sentry !== 'undefined' && Sentry.addBreadcrumb) {
                  Sentry.addBreadcrumb({
                    category: 'liveReleaseStale',
                    level: 'warning',
                    message: 'live-release sweep skipped title due to stale nextEpisode',
                    data: {
                      familyCode: familyDoc.id,
                      titleId: tdoc.id,
                      hasNextEpisode: !!t.nextEpisode,
                      refreshedAtAgeDays: refreshedAt ? Math.floor((now - refreshedAt) / 86400000) : null
                    }
                  });
                }
              } catch (_) {}
            }
            continue;
          }
          sweptCount++;

          const next = t.nextEpisode;
          if (!next || !next.airDate || next.season == null || next.episode == null) continue;

          // === D-09/D-10 — subscriber set: union-of-tuples ===
          const subscriberIds = new Set();
          for (const tk of Object.keys(t.tupleProgress)) {
            for (const mid of tk.split(',')) {
              if (mid) subscriberIds.add(mid);
            }
          }
          // === D-12 — per-show muted-by-member kill switch ===
          if (t.mutedShows) {
            for (const mid of Array.from(subscriberIds)) {
              if (t.mutedShows[mid]) subscriberIds.delete(mid);
            }
          }
          // === D-14 — ≥2 family members tracking the show ===
          if (subscriberIds.size < 2) continue;

          // === D-13 — 24h-before window (T-26h to T-23h, REVIEW LOW/MEDIUM-11 widened) ===
          // TMDB exposes air_date as DATE-ONLY (RESEARCH §Q6). 9pm best-guess
          // for window math; user-facing copy uses {day}-only.
          const airTs = new Date(next.airDate + 'T21:00:00').getTime();
          if (!Number.isFinite(airTs)) continue;
          const minsToAir = (airTs - now) / 60000;
          // Widened upper bound from 25h to 26h to absorb hourly tick-drift.
          if (minsToAir <= 60 * 23 || minsToAir > 60 * 26) continue;

          // === Idempotency — already fired (or suppressed) ===
          const epKey = `s${next.season}e${next.episode}`;
          if (t.liveReleaseFiredFor && t.liveReleaseFiredFor[epKey]) continue;

          // === D-16 — already-scheduled wins ===
          // REQUIRES composite index on (titleId asc, startAt asc) shipped in 15-01.
          let blocking = false;
          try {
            const wpSnap = await db.collection('families').doc(familyDoc.id)
              .collection('watchparties')
              .where('titleId', '==', tdoc.id)
              .where('startAt', '>=', airTs - 90 * 60000)
              .where('startAt', '<=', airTs + 90 * 60000)
              .get();
            for (const wpDoc of wpSnap.docs) {
              const wp = wpDoc.data();
              if (wp.status === 'archived' || wp.status === 'cancelled') continue;
              blocking = true;
              break;
            }
          } catch (e) {
            console.warn('[15-06] suppression query failed (index missing?)', familyDoc.id, tdoc.id, e.message);
            try {
              await tdoc.ref.update({ [`liveReleaseFiredFor.${epKey}`]: 'suppression_query_failed' });
            } catch (_) {}
            continue;
          }
          if (blocking) {
            suppressedCount++;
            try {
              await tdoc.ref.update({ [`liveReleaseFiredFor.${epKey}`]: 'suppressed_existing_wp' });
            } catch (e) { console.warn('[15-06] suppress flag write failed', e.message); }
            continue;
          }

          // === Fire the push ===
          // REVIEW MEDIUM-12 — "New episode tonight/tomorrow/{Weekday}" framing.
          const daysOut = Math.round((airTs - now) / 86400000);
          const when = daysOut === 0 ? 'tonight'
                     : daysOut === 1 ? 'tomorrow'
                     : new Date(airTs).toLocaleDateString('en-US', { weekday: 'long' });
          const showName = t.name || 'A show you watch';
          // Title: "New episode {when}". Body: "{Show} S{N}E{M} — watch with the couch?"
          const pushTitle = `New episode ${when}`;
          const pushBody = `${showName} S${next.season}E${next.episode} — watch with the couch?`;

          // Set the idempotency flag FIRST (flag-then-fire pattern).
          try {
            await tdoc.ref.update({ [`liveReleaseFiredFor.${epKey}`]: now });

            console.warn('[15-06] fire flag write failed; skipping push', e.message);
            continue;

          try {
            await sendToMembers(familyDoc.id, Array.from(subscriberIds), {
              title: pushTitle,
              body: pushBody,
              tag: `live-release-${tdoc.id}-${epKey}`,
              url: `/?nominate=${tdoc.id}&prefillTime=${airTs}`
            }, { eventType: 'newSeasonAirDate' });
            firedCount++;
          } catch (e) {
            console.warn('[15-06] live-release push failed', familyDoc.id, tdoc.id, epKey, e.message);
            // Flag stays set — accept push loss per D-15 cadence model.
          }
        }

        // === REVIEW HIGH-3 — update throttle cursor at end of sweep ===
        try {
          await familyDoc.ref.update({
            liveReleaseSweepLastRunAt: now,
            liveReleaseSweepStats: { sweptCount, staleCount, firedCount, suppressedCount, at: now }
          });
        } catch (e) {
          console.warn('[15-06] sweep cursor update failed', familyDoc.id, e.message);
          // If this fails, next tick will re-sweep — duplicate fires guarded by
          // liveReleaseFiredFor idempotency stamps anyway.
        }
