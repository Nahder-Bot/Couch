function renderPickerCard(game, picks, members, isTiebreaker) {
  if (!game) return '';
  const pickType = PICK_TYPE_BY_LEAGUE[game.league] || 'team_winner';
  const myMid = state.me.id;
  const myPick = picks.find(p => p.gameId === game.id && p.memberId === myMid);
  const allegiance = (state.participants && state.participants[myMid] && state.participants[myMid].teamAllegiance) || null;
  const isPreFilled = allegiance && (allegiance === game.homeTeam || allegiance === game.awayTeam);
  const isLocked = Date.now() >= game.startTime;
  const myTeamColor = (state.participants && state.participants[myMid] && state.participants[myMid].teamColor) || null;

  // Matchup line per UI-SPEC: "{Away} @ {Home}" (US sports) / "{Home} v {Away}" (soccer) / "{RaceName}" (F1) / "{Event}" (UFC)
  const matchupText = renderMatchupLine(game);
  const tiebreakerTag = isTiebreaker ? ' <em class="pe-tiebreaker-tag">(tiebreaker)</em>' : '';
  const meta = renderGameMeta(game);  // "Tonight, 7:30 PM" etc.

  // Render variant per pickType
  let chipRowHtml = '';
  switch (pickType) {
    case 'team_winner':         chipRowHtml = renderTeamWinnerChips(game, myPick, isPreFilled, myTeamColor, isLocked); break;
    case 'team_winner_or_draw': chipRowHtml = renderTeamWinnerOrDrawChips(game, myPick, isPreFilled, myTeamColor, isLocked); break;
    case 'f1_podium':           chipRowHtml = renderF1PodiumRows(game, myPick, isLocked); break;
    case 'ufc_winner_method':   chipRowHtml = renderUfcWinnerMethodChips(game, myPick, isLocked); break;
  }

  // D-09 pre-fill sub-line (only for team_winner / team_winner_or_draw)
  const prefillSubHtml = (isPreFilled && (pickType === 'team_winner' || pickType === 'team_winner_or_draw'))
    ? `<div class="pe-prefill-sub"><em>${escapeHtml(allegiance)}</em> - your pick from your team commitment. <em>Tap to change.</em></div>`
    : '';

  // D-05 tiebreaker numeric input
  const tiebreakerInputHtml = isTiebreaker
    ? `<div class="pe-tiebreaker-block">
         <label for="pe-tb-${escapeHtml(game.id)}">Predict total points (tiebreaker only)</label>
         <input id="pe-tb-${escapeHtml(game.id)}" class="pe-tiebreaker-input" type="number" inputmode="numeric" pattern="\\d*"
                placeholder="e.g. 47" value="${escapeHtml(myPick && myPick.tiebreakerTotal != null ? String(myPick.tiebreakerTotal) : '')}"
                ${isLocked ? 'disabled' : ''}
                onchange="onPickemTiebreakerChange('${escapeHtml(game.id)}', this.value)">
       </div>`
    : '';

  // Locked-state pill / countdown per UI-SPEC
  const lockHtml = renderLockState(game, isLocked);

  return `<article class="tab-list-card pe-picker-card" data-game-id="${escapeHtml(game.id)}">
    <div class="pe-matchup">${matchupText}${tiebreakerTag}</div>
    <div class="sports-game-time">${meta}</div>
    ${chipRowHtml}
    ${prefillSubHtml}
    ${tiebreakerInputHtml}
    ${lockHtml}
  </article>`;
}

function renderTeamWinnerChips(game, myPick, isPreFilled, myTeamColor, isLocked) {
  const pickedHome = myPick && myPick.selection && myPick.selection.winningTeam === 'home';
  const pickedAway = myPick && myPick.selection && myPick.selection.winningTeam === 'away';
  const myMid = state.me.id;
  const myAllegiance = (state.participants && state.participants[myMid] && state.participants[myMid].teamAllegiance) || null;
  const homeClass = ['pe-chip', pickedHome ? 'selected' : '', (isPreFilled && myAllegiance === game.homeTeam) ? 'prefilled' : '', isLocked ? 'locked' : ''].filter(Boolean).join(' ');
  const awayClass = ['pe-chip', pickedAway ? 'selected' : '', (isPreFilled && myAllegiance === game.awayTeam) ? 'prefilled' : '', isLocked ? 'locked' : ''].filter(Boolean).join(' ');
  const teamColorVar = myTeamColor ? `style="--team-color:${escapeHtml(myTeamColor)}"` : '';
  return `<div class="pe-chip-row" role="radiogroup">
    <button type="button" class="${homeClass}" ${teamColorVar} role="radio" aria-checked="${pickedHome}"
            ${isLocked ? 'disabled' : ''}
            onclick="onPickemSubmit('${escapeHtml(game.id)}', 'team_winner', {winningTeam:'home'})">
      ${escapeHtml(game.homeTeam)}
    </button>
    <button type="button" class="${awayClass}" ${teamColorVar} role="radio" aria-checked="${pickedAway}"
            ${isLocked ? 'disabled' : ''}
            onclick="onPickemSubmit('${escapeHtml(game.id)}', 'team_winner', {winningTeam:'away'})">
      ${escapeHtml(game.awayTeam)}
    </button>
  </div>`;
}

window.onPickemSubmit = async function(gameId, pickType, selection) {
  const myMid = state.me.id;
  const game = (state.pickemSlateGames || []).find(g => g.id === gameId);
  if (!game) return;
  const familyCode = state.familyCode;

  const pickDocId = await findOrCreatePickId(familyCode, myMid, gameId);  // existing or new (URL-safe random)
  const pickRef = db.collection('families').doc(familyCode).collection('picks').doc(pickDocId);

  if (!validatePickSelection({ pickType, selection }, game)) {
    flashToast("Invalid pick selection.");
    return;
  }

  try {
    const existing = await pickRef.get();

function renderInlineWpPickRow(wp, picks) {
  if (!wp || wp.mode !== 'game' || !wp.sportEvent) return '';
  if (Date.now() >= wp.sportEvent.startTime) return '';  // disappears at lock per D-07
  const gamePicks = (picks || []).filter(p => p.gameId === wp.sportEvent.id);
  if (gamePicks.length === 0) return '';  // hide when no member picked
  const myMid = state.me.id;
  const myPick = gamePicks.find(p => p.memberId === myMid);
  const peers = (state.members || []).filter(m => m.id !== myMid);
  const peerPicks = peers.map(m => {
    const p = gamePicks.find(pp => pp.memberId === m.id);
    return `${escapeHtml(m.name)}: ${p ? renderPickSummary(p) : '-'}`;
  }).join(' &middot; ');
  const myPart = myPick
    ? `You: ${renderPickSummary(myPick)}`
    : `<a href="#" onclick="openPickemForGame('${escapeHtml(wp.sportEvent.id)}'); return false;">You haven't picked yet - <em>make your pick</em></a>`;
  return `<div class="pe-inline-wp-row" aria-live="polite">
    <span class="pe-inline-wp-label">Picks &middot;</span>

  </div>`;
}
