      showScreen('pickem');
      const dayWindow = todayUtcDateString();
      const PICKEM_LEAGUES = leagueKeys.filter(k => k !== 'ufc');

      // Fetch schedules for active leagues in parallel.
      const schedulesByLeague = {};
      await Promise.all(PICKEM_LEAGUES.map(async lk => {
        try { schedulesByLeague[lk] = await fetchSchedule(lk, 7); }
        catch (e) { schedulesByLeague[lk] = []; }
      }));

      // Aggregate slate game IDs across all leagues.
      const slateGameIds = [];
      for (const lk of PICKEM_LEAGUES) {
        const games = schedulesByLeague[lk] || [];
        for (const g of games) if (g?.id) slateGameIds.push(g.id);
      }

      // REVIEWS Amendment 12 / MEDIUM-11 — Aggregate listener teardown.
      // Firestore 'in' query limit is 10; chunk slateGameIds. Each chunk gets its own
      // unsubscribe; aggregate function tears down all of them on screen change.
      // First teardown any prior listener.
      if (state.pickemPicksUnsubscribe) {
        try { state.pickemPicksUnsubscribe(); } catch (_) {}
        state.pickemPicksUnsubscribe = null;
      }

      const unsubs = [];
      const chunks = chunkArray(slateGameIds, 10);
      for (const chunk of chunks) {
        if (chunk.length === 0) continue;
        unsubs.push(
          db.collection('families').doc(state.familyCode).collection('picks')
            .where('gameId', 'in', chunk)
            .onSnapshot(snap => {
              snap.docChanges().forEach(change => {
                const pick = change.doc.data();
                if (!pick?.gameId || !pick?.memberId) return;
                if (!state.pickemPicksByGameMember[pick.gameId]) state.pickemPicksByGameMember[pick.gameId] = {};
                if (change.type === 'removed') {
                  delete state.pickemPicksByGameMember[pick.gameId][pick.memberId];
                } else {
                  state.pickemPicksByGameMember[pick.gameId][pick.memberId] = pick;
                }
              });
              renderPickemSurface(state.pickemActiveLeagueKey, dayWindow);
            })
        );
      }
      // AGGREGATE teardown function — single teardown handle for N chunked listeners.
      state.pickemPicksUnsubscribe = () => unsubs.forEach(fn => { try { fn(); } catch (_) {} });

      renderPickemSurface(null, dayWindow);
    }

    function chunkArray(arr, size) {
      const out = [];
      for (let i = 0; i < arr.length; i += size) out.push(arr.slice(i, i + size));
      return out;
    }
