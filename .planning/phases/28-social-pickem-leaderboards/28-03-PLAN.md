// ---- TheSportsDB score fetch — inline re-implementation per CJS/ESM boundary ----

async function fetchScoreInline(gameId, leagueKey) {
  if (!gameId) return null;
  const url = `https://www.thesportsdb.com/api/v1/json/1/lookupevent.php?id=${encodeURIComponent(gameId)}`;
  try {
    const fetch = (await import('node-fetch')).default;  // node-fetch already in queuenight deps
    const res = await fetch(url, { timeout: 10000 });
    if (!res.ok) return null;
    const data = await res.json();
    if (!data || !Array.isArray(data.events) || data.events.length === 0) return null;
    const ev = data.events[0];

    const status = (ev.strStatus || '').toLowerCase();
    const isFinal = status.includes('finished') || status.includes('ft') || status === 'match finished'
                    || (homeScore != null && awayScore != null && status === '');
    return {
      state: isFinal ? 'post' : (homeScore != null || awayScore != null ? 'in' : 'pre'),
      homeScore, awayScore,
      // Derive winningTeam for team_winner pickType
      winningTeam: (homeScore != null && awayScore != null)
        ? (homeScore > awayScore ? 'home' : (awayScore > homeScore ? 'away' : 'draw'))
        : null,
      // Derive result for team_winner_or_draw pickType
      result: (homeScore != null && awayScore != null)
        ? (homeScore > awayScore ? 'home' : (awayScore > homeScore ? 'away' : 'draw'))
        : null,
      strSeason: ev.strSeason || null
      // For f1_podium / ufc_winner_method: planner-discretion fallback —
      // free-tier TheSportsDB may not expose podium order or fight-method per
      // RESEARCH Risk 7 / A6 / A5. If finalGame.p1/.method absent at settlement,
      // scorePickInline returns 0 (default branch) — pick auto-zeros until
      // upstream provides the data. Acceptable v1 degradation per D-01.
    };
  } catch (e) {
    console.warn('gameResultsTick: fetchScoreInline failed', gameId, leagueKey, e.message);
    return null;
  }
}

exports.gameResultsTick = onSchedule({
  schedule: 'every 5 minutes',
  region: 'us-central1',
  timeoutSeconds: 240,
  memory: '256MiB'
}, async () => {
  const now = Date.now();
  let processed = 0;
  let settled = 0;
  let errored = 0;

  // Lazy-require sendToMembers (mirrors rsvpReminderTick pattern)
  let sendToMembers = null;
  try {
    const idx = require('../index.js');
    sendToMembers = idx.sendToMembers || null;
  } catch (e) {
    console.error('gameResultsTick: failed to require sendToMembers', e.message);
  }

  let families;
  try {
    families = await db.collection('families').get();
  } catch (e) {
    console.error('gameResultsTick: families.get() failed', e.message);
    return;
  }

  for (const familyDoc of families.docs) {
    const familyCode = familyDoc.id;
    let picksSnap;
    try {
      picksSnap = await db.collection('families').doc(familyCode)
        .collection('picks')
        .where('state', '==', 'pending')
        .where('gameStartTime', '<', now)
        .get();
    } catch (e) {
      console.warn('gameResultsTick: picks query failed', familyCode, e.message);
      errored++;
      continue;
    }

    // Group pending picks by gameId so we only fetch each score ONCE per tick
    const picksByGame = new Map();
    for (const pickDoc of picksSnap.docs) {
      const pick = pickDoc.data();
      if (pick.state === 'settled') continue;  // Pitfall 2 — short-circuit
      const key = `${pick.leagueKey}_${pick.gameId}`;
      if (!picksByGame.has(key)) picksByGame.set(key, []);
      picksByGame.get(key).push({ ref: pickDoc.ref, data: pick });
    }

    for (const [, gamePickGroup] of picksByGame) {
      const samplePick = gamePickGroup[0].data;
      processed++;
      const finalGame = await fetchScoreInline(samplePick.gameId, samplePick.leagueKey);
      if (!finalGame || finalGame.state !== 'post') continue;  // not done yet — retry next tick
      const actualTotal = (finalGame.homeScore != null && finalGame.awayScore != null)
        ? (finalGame.homeScore + finalGame.awayScore)
        : null;

      for (const { ref: pickRef, data: pick } of gamePickGroup) {
        try {
          const pointsAwarded = scorePickInline(pick, finalGame);
          const tiebreakerActual = pick.tiebreakerTotal != null ? actualTotal : null;
          const tiebreakerDelta = (pick.tiebreakerTotal != null && actualTotal != null)
            ? Math.abs(pick.tiebreakerTotal - actualTotal) : null;

          const lbRef = db.collection('families').doc(familyCode)
            .collection('leaderboards').doc(pick.leagueKey)
            .collection(pick.strSeason).doc('doc');

          let firedSeasonReset = false;
          await db.runTransaction(async (tx) => {
            // Re-read pick inside tx for atomic settle
            const pickSnap = await tx.get(pickRef);
            if (!pickSnap.exists || pickSnap.data().state === 'settled') return;  // raced — skip

            // Read current leaderboard doc
            const lbSnap = await tx.get(lbRef);
            const lb = lbSnap.exists ? lbSnap.data() : {
              leagueKey: pick.leagueKey,
              strSeason: pick.strSeason,
              members: {},
              updatedAt: now
            };
            const memberRow = lb.members[pick.memberId] || {
              pointsTotal: 0, picksTotal: 0, picksSettled: 0, picksAutoZeroed: 0,
              tiebreakerDeltaTotal: 0, tiebreakerCount: 0, lastPickAt: null
            };
            memberRow.pointsTotal += pointsAwarded;
            memberRow.picksTotal += 1;
            memberRow.picksSettled += 1;
            if (tiebreakerDelta != null) {
              memberRow.tiebreakerDeltaTotal += tiebreakerDelta;
              memberRow.tiebreakerCount += 1;
            }
            memberRow.lastPickAt = now;
            lb.members[pick.memberId] = memberRow;
            lb.updatedAt = now;
            tx.set(lbRef, lb, { merge: true });

            tx.update(pickRef, {
              state: 'settled',
              pointsAwarded,
              settledAt: admin.firestore.FieldValue.serverTimestamp(),
              tiebreakerActual,
              tiebreakerDelta
            });

            // D-12 season boundary detection: if this is the FIRST pick for this
            // (leagueKey, strSeason) AND a different strSeason exists for this
            // league in the family, fire pickemSeasonReset push (handled outside tx).
            if (!lbSnap.exists) {
              firedSeasonReset = true;  // signal post-tx push
            }
          });

          settled++;

          // Fire pickResults push (per-member) — outside tx
          if (sendToMembers) {
            try {
              await sendToMembers(familyCode, [pick.memberId], {
                title: 'Pick\'em results',
                body: `${pick.leagueKey.toUpperCase()} picks just settled. Tap to see how you did.`,
                tag: `pickem-results-${pick.memberId}-${pick.gameId}`,
                url: '/app?tab=pickem'
              }, {
                excludeMemberId: null,
                eventType: 'pickResults'
              });
            } catch (pushErr) {
              console.warn('gameResultsTick: pickResults push failed', familyCode, pick.memberId, pushErr.message);
            }
          }

          // D-12: fire pickemSeasonReset push if a new leaderboard doc was created
          // for this (leagueKey, strSeason) AND the family has prior seasons in this league
          if (firedSeasonReset && sendToMembers) {
            try {
              const otherSeasons = await db.collection('families').doc(familyCode)
                .collection('leaderboards').doc(pick.leagueKey).listCollections();
              if (otherSeasons.length > 1) {  // not the very first season ever
                await sendToMembers(familyCode, [pick.memberId], {
                  title: 'Pick\'em season reset',
                  body: `The ${pick.leagueKey.toUpperCase()} ${pick.strSeason} season just turned over. Fresh slate, fresh picks.`,
                  tag: `pickem-season-reset-${pick.leagueKey}-${pick.strSeason}`,
                  url: '/app?tab=pickem'
                }, {
                  excludeMemberId: null,
                  eventType: 'pickemSeasonReset'
                });
              }
            } catch (resetErr) {
              console.warn('gameResultsTick: pickemSeasonReset push failed', familyCode, pickErr ? pickErr.message : resetErr.message);
            }
          }
        } catch (e) {
          console.warn('gameResultsTick: per-pick failed', familyCode, pickRef.id, e.message);
          errored++;
        }
      }
    }
  }

  console.log(`gameResultsTick: processed=${processed} settled=${settled} errored=${errored}`);
});

// ---- TheSportsDB schedule fetch (inline per CJS/ESM boundary) ----
async function fetchUpcomingForLeague(leagueKey, leagueId, daysAhead = 1) {
  const url = `https://www.thesportsdb.com/api/v1/json/1/eventsnextleague.php?id=${leagueId}`;
  try {
    const fetch = (await import('node-fetch')).default;
    const res = await fetch(url, { timeout: 10000 });
    if (!res.ok) return [];
    const data = await res.json();
    if (!data || !Array.isArray(data.events)) return [];
    return data.events.map(ev => ({
      id: ev.idEvent,
      leagueKey,
      gameStartTime: ev.strTimestamp ? Date.parse(ev.strTimestamp) : null,
      strEvent: ev.strEvent || '',
      homeTeam: ev.strHomeTeam || '',
      awayTeam: ev.strAwayTeam || ''
    })).filter(g => g.gameStartTime != null && g.gameStartTime > Date.now());
  } catch (e) {
    console.warn('pickReminderTick: schedule fetch failed', leagueKey, e.message);
    return [];

}

  for (const [leagueKey, leagueId] of Object.entries(LEAGUE_IDS)) {
    const games = await fetchUpcomingForLeague(leagueKey, leagueId);
    for (const game of games) {
      const msUntilStart = game.gameStartTime - now;
      const inWindow = msUntilStart >= (PICK_REMINDER_OFFSET_MS - SLOP_MS)
                    && msUntilStart <= (PICK_REMINDER_OFFSET_MS + SLOP_MS);
      if (!inWindow) continue;

      // For each family, check who hasn't picked this game
      let families;
      try {
        families = await db.collection('families').get();
      } catch (e) {
        console.error('pickReminderTick: families.get() failed', e.message);
        errored++;
        continue;

        } catch (e) {
