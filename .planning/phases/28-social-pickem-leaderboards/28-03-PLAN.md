    function httpsGetJson(url) {
      return new Promise((resolve, reject) => {
        const req = https.get(url, { timeout: 10000 }, (res) => {
          let data = '';
          res.on('data', (chunk) => { data += chunk; });
          res.on('end', () => {
            if (res.statusCode >= 400) return reject(new Error(`HTTP ${res.statusCode}`));
            try { resolve(JSON.parse(data)); } catch (e) { reject(e); }
          });
        });
        req.on('error', reject);
        req.on('timeout', () => { req.destroy(); reject(new Error('timeout')); });
      });
    }

    async function fetchTsdScore(gameId, leagueKey) {
      const url = `${TSD_BASE}/${TSD_API_KEY}/lookupevent.php?id=${gameId}`;
      const data = await httpsGetJson(url);
      const ev = (data && data.events && data.events[0]) || null;
      if (!ev) return null;
      // Mirror js/sports-feed.js fetchScore + normalizeTsdEvent state mapping:
      const stateRaw = (ev.strStatus || '').toLowerCase();
      const isFinal = ['ft', 'aet', 'pen', 'final', 'match finished'].some(k => stateRaw.includes(k));
      if (!isFinal) return { state: 'in' };  // not yet final
      const homeScore = ev.intHomeScore != null && ev.intHomeScore !== '' ? parseInt(ev.intHomeScore, 10) : null;
      const awayScore = ev.intAwayScore != null && ev.intAwayScore !== '' ? parseInt(ev.intAwayScore, 10) : null;
      if (homeScore == null || awayScore == null) return { state: 'in' };  // can't resolve winner
      const winningTeam = homeScore > awayScore ? 'home' : (awayScore > homeScore ? 'away' : null);
      const result = winningTeam || 'draw';   // soccer: null winner → 'draw'
      return { state: 'post', homeScore, awayScore, winningTeam, result };
    }

    async function fetchJolpicaPodium(year, round) {
      const url = `${JOLPICA_BASE}/${year}/${round}/results/`;
      const data = await httpsGetJson(url);
      const races = (data && data.MRData && data.MRData.RaceTable && data.MRData.RaceTable.Races) || [];
      const results = (races[0] && races[0].Results) || [];
      if (results.length < 3) return { state: 'in' };  // race not yet complete
      const driverName = (d) => d ? `${d.givenName} ${d.familyName}` : '';
      const p1 = driverName(results[0] && results[0].Driver);
      const p2 = driverName(results[1] && results[1].Driver);
      const p3 = driverName(results[2] && results[2].Driver);
      if (!p1 || !p2 || !p3) return { state: 'in' };
      return { state: 'post', p1, p2, p3 };
    }

    exports.gameResultsTick = onSchedule({
      schedule: 'every 5 minutes',
      region: 'us-central1',
      timeoutSeconds: 240,
      memory: '256MiB'
    }, async () => {
      const now = Date.now();
      let processed = 0, settled = 0, autoZeroed = 0, errored = 0;

      let families;
      try { families = await db.collection('families').get(); }
      catch (e) { console.error('gameResultsTick: families list failed', e.message); return; }

      for (const familyDoc of families.docs) {
        const familyCode = familyDoc.id;
        let picksSnap;
        try {
          picksSnap = await db.collection('families').doc(familyCode)
            .collection('picks')
            .where('state', '==', 'pending')
            .where('gameStartTime', '<', now)
            .get();
        } catch (e) {
          console.warn('gameResultsTick picks query failed', familyCode, e.message);
          continue;
        }

        for (const pickDoc of picksSnap.docs) {
          processed++;
          const pick = pickDoc.data();
          try {
            // Idempotency short-circuit (D-16) — re-read race condition guard.
            if (pick.state === 'settled') continue;

            // Per-pickType score-fetch routing:
            let finalGame = null;
            if (pick.pickType === 'team_winner' || pick.pickType === 'team_winner_or_draw') {
              finalGame = await fetchTsdScore(pick.gameId, pick.leagueKey);
            } else if (pick.pickType === 'f1_podium') {
              // Parse year + round from pick.strSeason + pick.gameId or denormalized fields.
              // Plan 28-05 picker MUST stamp pick.f1Year + pick.f1Round at create time.
              if (!pick.f1Year || !pick.f1Round) {
                console.warn('gameResultsTick: f1_podium pick missing f1Year/f1Round', familyCode, pickDoc.id);
                continue;
              }
              finalGame = await fetchJolpicaPodium(pick.f1Year, pick.f1Round);
            } else {
              // Unknown pickType (e.g., legacy ufc_winner_method picks if any sneaked in) — skip.
              console.warn('gameResultsTick: unknown pickType skipped', familyCode, pickDoc.id, pick.pickType);
              continue;
            }

            // Auto-zero: if gameStartTime + grace passed AND we still cannot resolve a final score, auto-zero.
            if (!finalGame || finalGame.state !== 'post') {
              if (now > pick.gameStartTime + AUTO_ZERO_GRACE_MS) {
                await pickDoc.ref.update({
                  state: 'auto_zeroed',
                  pointsAwarded: 0,
                  settledAt: admin.firestore.FieldValue.serverTimestamp()
                });
                autoZeroed++;
              }
              continue;  // not yet final and not yet past grace — try next tick
            }

            // Score the pick.
            const pointsAwarded = scorePick(pick, finalGame);

            // Transactional pick-settle + leaderboard update (D-15 atomicity).
            const lbRef = db.collection('families').doc(familyCode)
              .collection('leaderboards').doc(`${pick.leagueKey}_${pick.strSeason}`);
            await db.runTransaction(async (tx) => {
              const lbSnap = await tx.get(lbRef);
              const lb = lbSnap.exists ? lbSnap.data() : {
                leagueKey: pick.leagueKey,
                strSeason: pick.strSeason,
                members: {}
              };
              const memberRow = lb.members[pick.memberId] || {
