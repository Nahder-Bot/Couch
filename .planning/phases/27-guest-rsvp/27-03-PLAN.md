**EDIT 7 — startStatusPoll:**
```js
    var statusPollHandle = null;
    function startStatusPoll() {
      if (statusPollHandle) return;
      statusPollHandle = setInterval(async function() {
        try {
          var r = await fetch(CF_BASE + '/rsvpStatus', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ data: { token: token, guestId: guestId } })
          });
          var d = await r.json();
          if (d && d.result) {
            if (d.result.revoked) {
              clearInterval(statusPollHandle); statusPollHandle = null;
              renderRevoked();
            } else if (d.result.closed) {
              clearInterval(statusPollHandle); statusPollHandle = null;
              renderClosed();
            } else if (d.result.expired) {
              clearInterval(statusPollHandle); statusPollHandle = null;
              renderExpired();
            }
          }
        } catch(e) {}
      }, 30000);
    }
```

    function urlBase64ToUint8Array(base64String) {
      var padding = '='.repeat((4 - base64String.length % 4) % 4);
      var base64 = (base64String + padding).replace(/-/g, '+').replace(/_/g, '/');
      var rawData = window.atob(base64);
      var outputArray = new Uint8Array(rawData.length);
      for (var i = 0; i < rawData.length; ++i) outputArray[i] = rawData.charCodeAt(i);
      return outputArray;
    }
    function replacePushBlock(stateClass, copyHtml) {
      var block = document.getElementById('rsvp-push-block');
      if (!block) return;
      block.outerHTML = '<div class="' + stateClass + '" role="status">' + copyHtml + '</div>';
    }
    window.requestPushOptIn = async function() {
      if (!('Notification' in window) || !('serviceWorker' in navigator) || !('PushManager' in window)) {
        replacePushBlock('rsvp-push-unsupported', "Reminders aren’t supported on this device — ask the host to message you when the party starts.");
        return;
      }
      try {
        var perm = await Notification.requestPermission();
        if (perm === 'denied') {
          replacePushBlock('rsvp-push-blocked', 'Notifications blocked in your browser settings.');
          return;
        }
        if (perm !== 'granted') {
          replacePushBlock('rsvp-push-unsupported', "Reminders aren’t supported on this device — ask the host to message you when the party starts.");
          return;
        }
        var reg;
        try {
          reg = await navigator.serviceWorker.register('/sw.js', { scope: '/' });
        } catch (swErr) {
          try { reg = await navigator.serviceWorker.register('/sw.js', { scope: '/rsvp/' }); }
          catch (e2) { throw swErr; }
        }
        await navigator.serviceWorker.ready;
        var sub = await reg.pushManager.subscribe({
          userVisibleOnly: true,
          applicationServerKey: urlBase64ToUint8Array(VAPID_PUBLIC_KEY)
        });
        var subJson = sub && sub.toJSON ? sub.toJSON() : null;
        if (!subJson || !subJson.endpoint) {
          replacePushBlock('rsvp-push-err', "Couldn’t set up reminders. Ask " + (lastHostName || 'the host') + ' to message you directly.');
          return;
        }
        try {
          var r = await fetch(CF_BASE + '/rsvpRevoke', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ data: { action: 'attachPushSub', token: token, guestId: guestId, pushSub: subJson } })
          });
          var data = await r.json();
          if (data && data.error) {
            replacePushBlock('rsvp-push-err', "Couldn’t set up reminders. Ask " + (lastHostName || 'the host') + ' to message you directly.');
            return;
          }
        } catch (postErr) {
          replacePushBlock('rsvp-push-err', "Couldn’t set up reminders. Ask " + (lastHostName || 'the host') + ' to message you directly.');
          return;
        }
        replacePushBlock('rsvp-push-done', "<em>You’re on the list.</em>");
      } catch (e) {
        replacePushBlock('rsvp-push-unsupported', "Reminders aren’t supported on this device — ask the host to message you when the party starts.");
      }
    };
