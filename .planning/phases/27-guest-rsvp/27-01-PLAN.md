```js
    const candidateDoc = await candidateRef.get();
    if (candidateDoc.exists) {
      wpRef = candidateRef;
      wpDoc = candidateDoc;
      break;
    }
  }
  if (!wpRef || !wpDoc) {
    throw new HttpsError('not-found', 'Watchparty not found.');
  }

  // Server-side host gate when auth context is present.
  const wp = wpDoc.data() || {};
  const callerUid = request.auth && request.auth.uid;
  if (callerUid && wp.hostUid && callerUid !== wp.hostUid) {
    throw new HttpsError('permission-denied', 'Only the host can revoke or close RSVPs.');
  }
  if (callerUid && !wp.hostUid && wp.hostId && callerUid !== wp.hostId) {
    // Legacy wps without hostUid — fall back to hostId. Mismatch denied.
    throw new HttpsError('permission-denied', 'Only the host can revoke or close RSVPs.');
  }

  if (action === 'close') {
    await wpRef.update({ rsvpClosed: true });
    return { ok: true, action: 'close' };
  }

  // action === 'revoke' — soft-delete via runTransaction (RESEARCH Q2 array upsert pattern).
  await db.runTransaction(async (tx) => {
    const snap = await tx.get(wpRef);
    if (!snap.exists) throw new HttpsError('not-found', 'Watchparty disappeared.');
    const data = snap.data() || {};
    const guests = Array.isArray(data.guests) ? data.guests.slice() : [];
    const idx = guests.findIndex(g => g && g.guestId === guestId);
    if (idx === -1) {
      // Idempotent: already-removed guest returns ok without re-write.
      return;
    }
    guests[idx] = Object.assign({}, guests[idx], { revoked: true });
    const visible = guests.filter(g => g && !g.revoked);
    tx.update(wpRef, {
      guests: guests,
      guestCount: visible.length
    });
  });

  return { ok: true, action: 'revoke', guestId: guestId };
});
```
