let _wpFlipTimers = new Map();  // wpId → timeoutId

function scheduleWatchpartyFlips(parties) {
  // Clear timers for parties no longer scheduled
  const current = new Set(parties.filter(p => p.status === 'scheduled').map(p => p.id));
  for (const [id, tid] of _wpFlipTimers.entries()) {
    if (!current.has(id)) { clearTimeout(tid); _wpFlipTimers.delete(id); }
  }
  const now = Date.now();
  for (const wp of parties) {
    if (wp.status !== 'scheduled') continue;
    if (_wpFlipTimers.has(wp.id)) continue;
    const delay = Math.max(0, (wp.startAt || 0) - now);
    if (delay > 6 * 60 * 60 * 1000) continue;  // CF handles far-future
    const tid = setTimeout(() => flipWatchpartyToActive(wp.id), delay);
    _wpFlipTimers.set(wp.id, tid);
  }
}

async function flipWatchpartyToActive(wpId) {
  _wpFlipTimers.delete(wpId);
  try {
    const ref = watchpartyRef(wpId);
    const snap = await getDoc(ref);
    if (!snap.exists()) return;
    const wp = snap.data();
    if (wp.status !== 'scheduled') return;  // already flipped (by another client or CF)
    if ((wp.startAt || 0) > Date.now()) return;  // defensive
    await updateDoc(ref, { status: 'active', lastActivityAt: Date.now(), ...writeAttribution() });
  } catch(e) { qnLog('[wp] flip failed', e.message); }
}

const { onSchedule } = require('firebase-functions/v2/scheduler');

exports.watchpartyTick = onSchedule({ schedule: 'every 5 minutes', region: 'us-central1' }, async () => {
  const now = Date.now();
  const SIX_HOURS = 6 * 60 * 60 * 1000;
  const THIRTY_MIN = 30 * 60 * 1000;
  // Iterate every family. Alternative would be a collection-group query, but we need per-family paths.
  const families = await db.collection('families').get();
  for (const familyDoc of families.docs) {
    const wpSnap = await db.collection('families').doc(familyDoc.id).collection('watchparties').get();
    for (const doc of wpSnap.docs) {
      const wp = doc.data();
      // Branch 1: flip scheduled → active if past startAt
      if (wp.status === 'scheduled' && (wp.startAt || 0) <= now) {
        // But not if too stale — those go to archived
        if ((wp.startAt || 0) < now - SIX_HOURS) {
          await doc.ref.update({ status: 'archived', archivedAt: now, archivedReason: 'stale_scheduled' });
        } else {
          await doc.ref.update({ status: 'active', lastActivityAt: now });
        }
        continue;
      }
      // Branch 2: archive empty active parties
      if (wp.status === 'active') {
        const participantCount = Object.keys(wp.participants || {}).length;
        const lastActivity = wp.lastActivityAt || wp.startAt || 0;
        if (participantCount === 0 && lastActivity < now - THIRTY_MIN) {
          await doc.ref.update({ status: 'archived', archivedAt: now, archivedReason: 'empty_active_timeout' });
        }
      }
    }
  }
  return null;
});
