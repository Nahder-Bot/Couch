<!-- "+ more" button opens iOS native emoji keyboard via hidden-input focus trick.
     No JS emoji-picker library (keeps single-file-no-bundler constraint from CLAUDE.md).
     Flow:
       1. User taps "+" button
       2. Hidden <input id="wp-emoji-input"> receives focus → native keyboard opens
       3. iOS: user taps globe to switch to emoji tab (user may need to do this once;
          system remembers the last-used keyboard thereafter)
       4. oninput fires with the typed char
       5. We extract the first grapheme cluster (to handle multi-codepoint emoji like 👨‍👩‍👧‍👦)
       6. sendReaction(char) via the same path as palette taps
       7. Clear + blur input
-->
<!--
<input id="wp-emoji-input" type="text" autocomplete="off" autocapitalize="none"
  style="position:fixed;left:-9999px;top:-9999px;width:1px;height:1px;opacity:0;" aria-hidden="true">

function openEmojiPicker() {
  const input = document.getElementById('wp-emoji-input');
  if (!input) return;
  input.value = '';
  input.focus();
  input.oninput = () => {
    const val = input.value;
    if (!val) return;
    // Grapheme-cluster-safe first-char extraction for multi-codepoint emoji
    const first = Array.from(new Intl.Segmenter('en', { granularity: 'grapheme' }).segment(val))[0]?.segment;
    if (first) sendReaction(first);
    input.value = '';
    input.blur();
  };
}
-->
