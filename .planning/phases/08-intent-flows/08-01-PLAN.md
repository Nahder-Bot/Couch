match /families/{familyCode}/intents/{intentId} {
  allow read: if isMemberOfFamily(familyCode);
  allow create: if isMemberOfFamily(familyCode)
    && request.resource.data.createdBy == memberIdOf(familyCode)
    && request.resource.data.status == 'open';
  // Update: allow RSVP writes by any family member on their own rsvps slot;
  // allow status/cancelledAt writes only by the creator;
  // allow status='matched' transitions by any family member (match auto-detect).
  allow update: if isMemberOfFamily(familyCode) && (
    // RSVP branch: only rsvps.{myMemberId} + writeAttribution fields touched
    request.resource.data.diff(resource.data).affectedKeys()
      .hasOnly(['rsvps','actingUid','memberId','memberName'])
    ||
    // Cancel branch: only the creator can cancel
    (resource.data.createdBy == memberIdOf(familyCode) &&
     request.resource.data.status in ['cancelled'] &&
