async function maybeEvaluateIntentMatches(intents) {

    const { required, yesCount, eligibleCount } = computeIntentTally(intent);

    // Transition to matched — idempotent under races
    try {
      const snap = await getDoc(intentRef(intent.id));
      if (!snap.exists()) continue;
      const fresh = snap.data();
      if (fresh.status !== 'open') continue;
      await updateDoc(intentRef(intent.id), {
        status: 'matched',
        matchedAt: Date.now(),
        ...writeAttribution()
      });
    } catch(e) { qnLog('[intent] match write failed', intent.id, e.message); }
  }

  // Separately: for intents freshly-transitioned to matched AND where I'm the creator, prompt

function computeIntentTally(intent) {
  const eligibleCount = state.members.filter(m => !m.temporary || (m.temporary && (m.expiresAt||0) > Date.now())).length;
  const rsvps = intent.rsvps || {};
  const yesCount = Object.values(rsvps).filter(r => r.value === 'yes').length;
  const rule = intent.thresholdRule || 'any_yes';
  let required;
  if (rule === 'majority') required = Math.ceil(eligibleCount / 2);
  else required = Math.max(2, 2); // any_yes → 2 (creator + at least 1 co-signer; creator's own is always counted)
  return { required, yesCount, eligibleCount, rule };
}
