    ```javascript
    #!/usr/bin/env node
    // Phase 12 / POL-03 + POL-04 — One-time TMDB id audit.
    // Reads COUCH_NIGHTS_PACKS from js/constants.js, fetches each id from TMDB,
    // and writes a plain-text PASS/MISMATCH report. Run from repo root.
    // Self-deletes in a finally block — even on partial run, the file is gone after exit.

    const fs = require('fs');
    const path = require('path');

    const REPO_ROOT = path.resolve(__dirname, '..', '..', '..');
    const CONSTANTS_PATH = path.join(REPO_ROOT, 'js', 'constants.js');
    const REPORT_PATH = path.join(__dirname, 'audit-report.txt');

    async function main() {
      const src = fs.readFileSync(CONSTANTS_PATH, 'utf8');

      const keyMatch = src.match(/export const TMDB_KEY\s*=\s*['"]([^'"]+)['"]/);
      if (!keyMatch) { console.error('Could not extract TMDB_KEY from js/constants.js'); process.exit(1); }
      const TMDB_KEY = keyMatch[1];

      // Extract COUCH_NIGHTS_PACKS array region.
      const startIdx = src.indexOf('export const COUCH_NIGHTS_PACKS');
      if (startIdx < 0) { console.error('Could not find COUCH_NIGHTS_PACKS'); process.exit(1); }
      // Find the matching `];` end of the array.
      const endIdx = src.indexOf('];', startIdx);
      if (endIdx < 0) { console.error('Could not find end of COUCH_NIGHTS_PACKS'); process.exit(1); }
      const region = src.slice(startIdx, endIdx + 2);

      // EXPECTED_COUNT = total integer literals inside any tmdbIds: [...] block in the region.
      // We deliberately scope to tmdbIds blocks to exclude the heroImageUrl URL paths and other digits.
      const tmdbBlockRegex = /tmdbIds:\s*\[([^\]]+)\]/g;
      let expectedCount = 0;
      let bm;
      while ((bm = tmdbBlockRegex.exec(region)) !== null) {
        const ids = bm[1].match(/\b\d+\b/g) || [];
        expectedCount += ids.length;
      }

      // Parse: split into pack objects via the `id: 'foo-bar',` lines.
      const packIdRegex = /id:\s*['"]([a-z0-9-]+)['"]/g;
      const packs = [];
      let m;
      while ((m = packIdRegex.exec(region)) !== null) {
        const packId = m[1];
        const slice = region.slice(m.index, m.index + 2000);
        const tmdbBlock = slice.match(/tmdbIds:\s*\[([^\]]+)\]/);
        if (!tmdbBlock) continue;
        const idsRaw = tmdbBlock[1];
        const ids = idsRaw.match(/\d+/g).map(s => parseInt(s, 10));
        const beforeBlock = slice.slice(0, slice.indexOf('tmdbIds:'));
        const commentMatches = beforeBlock.match(/\/\/[^\n]+/g) || [];
        const lastTwoComments = commentMatches.slice(-2).map(c => c.replace(/^\/\/\s*/, '').trim()).join(' ');
        const labels = lastTwoComments.split(',').map(l => l.trim()).filter(Boolean);
        packs.push({ packId, ids, labels });
      }

      function sleep(ms) { return new Promise(r => setTimeout(r, ms)); }

      async function fetchMovie(id) {
        const url = `https://api.themoviedb.org/3/movie/${id}?api_key=${TMDB_KEY}`;
        const r = await fetch(url);
        if (!r.ok) return { ok: false, status: r.status, id };
        const j = await r.json();
        return { ok: true, id, title: j.title, originalTitle: j.original_title, releaseDate: j.release_date };
      }

      function normalize(s) {
        return (s || '').toLowerCase().replace(/[^a-z0-9 ]/g, '').replace(/\s+/g, ' ').trim();
      }

      const lines = [];
      lines.push('# COUCH_NIGHTS_PACKS — TMDB ID Audit Report');
      lines.push('# Phase 12 / POL-03 + POL-04');
      lines.push('# Generated: ' + new Date().toISOString());
      lines.push('EXPECTED_COUNT: ' + expectedCount);
      lines.push('');

      let totalPass = 0, totalMismatch = 0, totalNotFound = 0;

      for (const pack of packs) {
        lines.push('=== ' + pack.packId + ' ===');
        for (let i = 0; i < pack.ids.length; i++) {
          const id = pack.ids[i];
          const expectedLabel = pack.labels[i] || '(no label)';
          const r = await fetchMovie(id);
          await sleep(200);
          if (!r.ok) {
            lines.push(`  ${id}\tNOT_FOUND (HTTP ${r.status})\texpected: "${expectedLabel}"`);
            totalNotFound++;
            continue;
          }
          const titleNorm = normalize(r.title);
          const labelNorm = normalize(expectedLabel);
          const match = titleNorm === labelNorm
            || titleNorm.includes(labelNorm)
            || labelNorm.includes(titleNorm);
          const status = match ? 'PASS' : 'MISMATCH';
          if (match) totalPass++; else totalMismatch++;
          lines.push(`  ${id}\t${status}\tactual: "${r.title}" (${(r.releaseDate||'').slice(0,4)})\texpected: "${expectedLabel}"`);
        }
        lines.push('');
      }

      lines.push('=== SUMMARY ===');
      lines.push(`Total packs: ${packs.length}`);
      lines.push(`Total ids audited: ${totalPass + totalMismatch + totalNotFound}`);
      lines.push(`PASS: ${totalPass}`);
      lines.push(`MISMATCH: ${totalMismatch}`);
      lines.push(`NOT_FOUND: ${totalNotFound}`);

      fs.writeFileSync(REPORT_PATH, lines.join('\n'), 'utf8');
      console.log('Wrote ' + REPORT_PATH);
      console.log('EXPECTED_COUNT=' + expectedCount + ' PASS=' + totalPass + ' MISMATCH=' + totalMismatch + ' NOT_FOUND=' + totalNotFound);
    }

    (async () => {
      try {
        await main();
      } catch (e) {
        console.error('Audit script failed:', e && e.stack || e);
        process.exitCode = 1;
      } finally {
        // Self-delete — script is a one-time tool, not a checked-in artifact.
        // Even on partial run / thrown error, the file is gone after exit.
        try { fs.unlinkSync(__filename); } catch (_) {}
      }
    })();
    ```
